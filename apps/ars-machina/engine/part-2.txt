function redo() {
  if (!state.redo.length || state.mode !== 'build') return;
  state.undo.push(snapshot());
  const model = JSON.parse(state.redo.pop());
  applyModel(model);
  saveSoon();
  updateHistoryButtons();
  clickSound(330, .035);
}

function updateHistoryButtons() {
  ui.undoButton.disabled = !state.undo.length || state.mode !== 'build';
  ui.redoButton.disabled = !state.redo.length || state.mode !== 'build';
}

function resetVelocities() {
  syncAllMounts(true);
  for (const node of state.nodes) {
    node.px = node.x;
    node.py = node.y;
  }
  for (const rotor of state.rotors) rotor.omega = 0;
}

function renderTools() {
  ui.toolRail.innerHTML = '';
  for (const [tool, label, icon] of TOOL_GROUPS[state.category]) {
    const button = document.createElement('button');
    button.type = 'button';
    button.className = `tool${state.tool === tool ? ' active' : ''}`;
    button.dataset.tool = tool;
    button.dataset.nativePress = '';
    button.innerHTML = `<span class="tool-icon icon-${icon}" aria-hidden="true"></span><span class="tool-label">${label}</span>`;
    button.addEventListener('click', () => setTool(tool));
    ui.toolRail.append(button);
  }
}

function setCategory(category) {
  state.category = category;
  document.querySelectorAll('.category').forEach((button) => button.classList.toggle('active', button.dataset.category === category));
  if (!TOOL_GROUPS[category].some(([tool]) => tool === state.tool)) state.tool = TOOL_GROUPS[category][0][0];
  renderTools();
}

function updateModeUI() {
  const running = state.mode === 'run';
  ui.buildDock.hidden = running;
  ui.runDock.hidden = !running;
  ui.modeLabel.textContent = running ? 'испытательный двор' : 'чертёжный двор';
  ui.blueprintsButton.disabled = running;
  document.querySelector('.history-controls').hidden = running;
  ui.inspector.hidden = running || !state.selection;
  updateHistoryButtons();
  if (running) setYardNote('Механизм ожил. Привод, лебёдка и спуск — внизу.', 2200);

function enterRunMode() {
  if (state.mode === 'run') return;
  saveNow();
  state.initialSnapshot = snapshot();
  state.mode = 'run';
  state.pendingNodeId = null;
  state.selection = null;
  state.drive = 0;
  state.simTime = 0;
  fireCountdown = -1;
  activeWinchDirection = 0;
  resetVelocities();
  updateDriveButtons();
  updateModeUI();
  heavySound();
}

function returnToBuild() {
  if (state.mode !== 'run') return;
  try {
    applyModel(JSON.parse(state.initialSnapshot || snapshot()));
  } catch {
    state.mode = 'build';
  }
  state.initialSnapshot = null;
  state.mode = 'build';
  updateModeUI();
  setYardNote('Испытание завершено. Чертёж восстановлен.', 1800);
  clickSound(210, .05);
}

function triggerFire() {
  if (state.mode !== 'run') return;
  let count = 0;
  for (const constraint of state.constraints) {
    if (constraint.type === 'latch' && constraint.active) {
      constraint.active = false;
      count += 1;
    }
  }
  const payloadLocks = state.constraints.filter((item) => item.type === 'payloadLock' && item.active);
  if (payloadLocks.length) fireCountdown = count ? .62 : .05;
  ui.fireButton.classList.add('active');
  setTimeout(() => ui.fireButton.classList.remove('active'), 180);
  if (count || payloadLocks.length) {
    snapSound();
    vibrate([18, 24, 28]);
    setYardNote(count ? 'Защёлка открыта. Снаряд удержится до замаха.' : 'Снаряд освобождён.', 1300);
  } else {
    setYardNote('В конструкции нет заряженной защёлки.', 1400);
    clickSound(130, .04);
  }
}

function releasePayloads() {
  let released = 0;
  for (const constraint of state.constraints) {
    if (constraint.type === 'payloadLock' && constraint.active) {
      constraint.active = false;
      released += 1;
    }
  }
  if (released) {
    heavySound(520, .09);
    setYardNote('Снаряд свободен.', 900);
  }
}

function ensureAudio() {
  if (!prefs.sound || firstInteraction) return null;
  if (!audioContext) {
    const AudioCtor = window.AudioContext || window.webkitAudioContext;
    if (!AudioCtor) return null;
    audioContext = new AudioCtor();
  }
  if (audioContext.state === 'suspended') audioContext.resume().catch(() => {});
  return audioContext;
}

function clickSound(frequency = 260, duration = .035) {
  const ac = ensureAudio();
  if (!ac) return;
  const oscillator = ac.createOscillator();
  const gain = ac.createGain();
  oscillator.type = 'triangle';
  oscillator.frequency.setValueAtTime(frequency, ac.currentTime);
  oscillator.frequency.exponentialRampToValueAtTime(Math.max(60, frequency * .72), ac.currentTime + duration);
  gain.gain.setValueAtTime(.035, ac.currentTime);
  gain.gain.exponentialRampToValueAtTime(.0001, ac.currentTime + duration);
  oscillator.connect(gain).connect(ac.destination);
  oscillator.start();
  oscillator.stop(ac.currentTime + duration);
}

function heavySound(frequency = 115, duration = .12) {
  const ac = ensureAudio();
  if (!ac) return;
  const oscillator = ac.createOscillator();
  const gain = ac.createGain();
  oscillator.type = 'sawtooth';
  oscillator.frequency.setValueAtTime(frequency, ac.currentTime);
  oscillator.frequency.exponentialRampToValueAtTime(55, ac.currentTime + duration);
  gain.gain.setValueAtTime(.04, ac.currentTime);
  gain.gain.exponentialRampToValueAtTime(.0001, ac.currentTime + duration);
  oscillator.connect(gain).connect(ac.destination);
  oscillator.start();
  oscillator.stop(ac.currentTime + duration);
}

function worldToScreen(x, y) {
  return {
    x: (x - state.camera.x) * state.camera.zoom + width / 2,
    y: (y - state.camera.y) * state.camera.zoom + height / 2
  };
}

function screenToWorld(x, y) {
  return {
    x: (x - width / 2) / state.camera.zoom + state.camera.x,
    y: (y - height / 2) / state.camera.zoom + state.camera.y
  };
}

function canvasPoint(event) {
  const rect = canvas.getBoundingClientRect();
  return { x: event.clientX - rect.left, y: event.clientY - rect.top };
}

function resizeCanvas() {
  const rect = canvas.getBoundingClientRect();
  width = Math.max(1, rect.width);
  height = Math.max(1, rect.height);
  dpr = Math.min(window.devicePixelRatio || 1, 2);
  const pixelWidth = Math.round(width * dpr);
  const pixelHeight = Math.round(height * dpr);
  if (canvas.width !== pixelWidth || canvas.height !== pixelHeight) {
    canvas.width = pixelWidth;
    canvas.height = pixelHeight;
  }
  ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
}

function findNodeAt(world, radius = 18 / state.camera.zoom, { includeMounts = true } = {}) {
  let best = null;
  let bestDistance = radius;
  for (const node of state.nodes) {
    if (!includeMounts && node.mount) continue;
    const d = Math.hypot(node.x - world.x, node.y - world.y);
    if (d < bestDistance || (Math.abs(d - bestDistance) < .01 && itemLayer(node) > itemLayer(best))) {
      best = node;
      bestDistance = d;
    }
  }
  return best;
}

function distanceToSegment(point, a, b) {
  const vx = b.x - a.x;
  const vy = b.y - a.y;
  const wx = point.x - a.x;
  const wy = point.y - a.y;
  const lengthSquared = vx * vx + vy * vy;
  const t = lengthSquared ? clamp((wx * vx + wy * vy) / lengthSquared, 0, 1) : 0;
  return Math.hypot(point.x - (a.x + vx * t), point.y - (a.y + vy * t));
}

function findThingsAt(world) {
  const candidates = [];
  for (const joint of state.joints) {
    const a = nodeById(joint.a);
    const b = nodeById(joint.b);
    if (!a || !b) continue;
    const point = { x: (a.x + b.x) / 2, y: (a.y + b.y) / 2 };
    const d = distance(world, point);
    if (d <= 20 / state.camera.zoom) candidates.push({ selection: { kind: 'joint', id: joint.id }, distance: d, layer: itemLayer(joint), priority: 5 });
  }
  for (const rotor of state.rotors) {
    const node = nodeById(rotor.nodeId);
    if (!node) continue;
    const d = Math.abs(Math.hypot(world.x - node.x, world.y - node.y) - rotor.radius);
    if (d < Math.max(12, rotor.radius * .35)) candidates.push({ selection: { kind: 'rotor', id: rotor.id }, distance: d, layer: itemLayer(rotor), priority: 4 });
  }
  for (const attachment of state.attachments) {
    const node = nodeById(attachment.nodeId);
    if (!node) continue;
    const d = distance(world, node);
    if (d <= 22 / state.camera.zoom) candidates.push({ selection: { kind: 'attachment', id: attachment.id }, distance: d, layer: itemLayer(attachment), priority: 4 });
  }
  for (const node of state.nodes) {
    const d = distance(world, node);
    if (d <= 20 / state.camera.zoom) candidates.push({ selection: { kind: 'node', id: node.id }, distance: d, layer: itemLayer(node), priority: node.mount ? 5 : 3 });
  }
  for (const constraint of state.constraints) {
    if (!constraint.active && state.mode === 'run') continue;
    const a = nodeById(constraint.a);
    const b = nodeById(constraint.b);
    if (!a || !b) continue;
    const d = distanceToSegment(world, a, b);
    if (d <= 12 / state.camera.zoom) candidates.push({ selection: { kind: 'constraint', id: constraint.id }, distance: d, layer: itemLayer(constraint), priority: constraint.type === 'beam' ? 2 : 1 });
  }
  return candidates.sort((left, right) => right.layer - left.layer || right.priority - left.priority || left.distance - right.distance);
}

function findThingAt(world) {
  return findThingsAt(world)[0]?.selection || null;
}

function cycleSelectionAt(world) {
  const candidates = findThingsAt(world);
  if (!candidates.length) {
    state.hitCycle = null;
    selectThing(null);
    return null;
  }
  const signature = candidates.map((candidate) => `${candidate.selection.kind}:${candidate.selection.id}`).join('|');
  const close = state.hitCycle && distance(state.hitCycle.world, world) < 18 / state.camera.zoom && state.hitCycle.signature === signature;
  const index = close ? (state.hitCycle.index + 1) % candidates.length : 0;
  state.hitCycle = { world: { ...world }, signature, index };
  const selection = candidates[index].selection;
  selectThing(selection);
  setYardNote(`${index + 1}/${candidates.length}: ${selectionDisplayName(selection)}`, 1200);
  return selection;
}

function selectThing(selection) {
  state.selection = selection;
  updateInspector();
}

function getSelectedFor(selection) {
  if (!selection) return null;
  if (selection.kind === 'node') return nodeById(selection.id);
  if (selection.kind === 'rotor') return rotorById(selection.id);
  if (selection.kind === 'attachment') return attachmentById(selection.id);
  if (selection.kind === 'constraint') return constraintById(selection.id);
  if (selection.kind === 'joint') return jointById(selection.id);
  return null;
}

function selectionDisplayName(selection) {
  const selected = getSelectedFor(selection);
  if (!selected) return 'пусто';
  let type = selected.type || 'node';
  if (selection.kind === 'node') type = selected.mount ? 'mount' : selected.role === 'payload' ? 'payload' : 'node';
  return (NAMES[type] || ['Деталь', type])[1];
}

function selectionWorld(selection = state.selection) {
  const selected = getSelectedFor(selection);
  if (!selected) return null;
  if (selection.kind === 'node') return { x: selected.x, y: selected.y };
  if (selection.kind === 'rotor' || selection.kind === 'attachment') {
    const node = nodeById(selected.nodeId);
    return node ? { x: node.x, y: node.y } : null;
  }
  if (selection.kind === 'constraint') {
    const a = nodeById(selected.a);
    const b = nodeById(selected.b);
    return a && b ? { x: (a.x + b.x) / 2, y: (a.y + b.y) / 2 } : null;
  }
  if (selection.kind === 'joint') {
    const a = nodeById(selected.a);
    const b = nodeById(selected.b);
    return a && b ? { x: (a.x + b.x) / 2, y: (a.y + b.y) / 2 } : null;
  }
  return null;
}

function adjustLayer(direction) {
  const selected = getSelected();
  if (!selected) return;
  pushUndo();
  selected.layer = clamp(itemLayer(selected) + direction, -8, 8);
  ui.inspectLayerLabel.textContent = `слой ${selected.layer}`;
  saveSoon();
  clickSound(direction > 0 ? 410 : 180, .03);
}

function cycleSelected() {
  const point = selectionWorld();
  if (point) cycleSelectionAt(point);
}

function updateInspector() {
  if (!state.selection || state.mode !== 'build') {
    ui.inspector.hidden = true;
    return;
  }
  const selected = getSelected();
  if (!selected) {
    state.selection = null;
    ui.inspector.hidden = true;
    return;
  }
  ui.inspector.hidden = false;
  let type = selected.type || 'node';
  if (state.selection.kind === 'node') type = selected.mount ? 'mount' : selected.role === 'payload' ? 'payload' : 'node';
  const [kind, name] = NAMES[type] || ['Деталь', type];
  ui.inspectorKind.textContent = kind;
  ui.inspectorName.textContent = name;
  ui.inspectValue.textContent = inspectorValue(selected, type);
  ui.inspectLayerLabel.textContent = `слой ${itemLayer(selected)}`;
  const adjustable = state.selection.kind !== 'constraint' || selected.type === 'spring';
  ui.inspectMinus.disabled = !adjustable && !selected.mount;
  ui.inspectPlus.disabled = !adjustable && !selected.mount;
}

function getSelected() {
  return getSelectedFor(state.selection);
}

function inspectorValue(selected, type) {
  if (state.selection.kind === 'joint') return type === 'weld' ? 'угол зафиксирован' : 'свободное вращение';
  if (state.selection.kind === 'rotor') return `Ø ${Math.round(selected.radius * 2)}`;
  if (state.selection.kind === 'node') {
    if (selected.mount) return `${Math.round(selected.mount.t * 100)}% балки`;
    return selected.fixed ? 'закреплён' : `масса ${selected.mass.toFixed(1)}`;
  }
  if (state.selection.kind === 'attachment') {
    if (type === 'motor') return `сила ${selected.amount.toFixed(1)}`;
    if (type === 'weight') return `масса ${selected.amount.toFixed(0)}`;
    return 'заряжен';
  }
  if (state.selection.kind === 'constraint' && type === 'spring') return `длина ${Math.round(selected.length)}`;
  return `длина ${Math.round(selected.length || 0)}`;
}

function translateBeam(beam, dx, dy) {
  if (!beam) return false;
  const endpoints = [nodeById(beam.a), nodeById(beam.b)];
  if (endpoints.some((node) => !node || node.fixed || node.mount)) return false;
  for (const node of endpoints) {
    node.x += dx;
    node.y += dy;
    node.px += dx;
    node.py += dy;
  }
  syncAllMounts(true);
  return true;
}

function moveMountAlongBeam(node, deltaT) {
  if (!node?.mount) return false;
  const host = beamById(node.mount.beamId);
  if (!host) return false;
  const nextT = clamp(node.mount.t + deltaT, .03, .97);
  const a = nodeById(host.a);
  const b = nodeById(host.b);
  const nextPoint = { x: a.x + (b.x - a.x) * nextT, y: a.y + (b.y - a.y) * nextT };
  const joint = state.joints.find((item) => item.a === node.id || item.b === node.id);
  if (joint) {
    const peerId = joint.a === node.id ? joint.b : joint.a;
    if (peerId === node.id) {
      node.mount.t = nextT;
      syncAllMounts(true);
      return true;
    }
    const peer = nodeById(peerId);
    const peerBeam = peer?.mount ? beamById(peer.mount.beamId) : null;
    if (peerBeam) {
      const delta = { x: nextPoint.x - peer.x, y: nextPoint.y - peer.y };
      if (!translateBeam(peerBeam, delta.x, delta.y)) return false;
    } else if (peer && !peer.fixed) {
      peer.x = nextPoint.x;
      peer.y = nextPoint.y;
      peer.px = peer.x;
      peer.py = peer.y;
    }
  }
  node.mount.t = nextT;
  syncAllMounts(true);
  return true;
}

function adjustSelection(direction) {
  const selected = getSelected();
  if (!selected) return;
  pushUndo();
  let changed = true;
  if (state.selection.kind === 'rotor') {
    selected.radius = clamp(selected.radius + direction * 6, 14, 90);
    selected.teeth = Math.max(8, Math.round(selected.radius / 3));
  } else if (state.selection.kind === 'node' && selected.mount) {
    changed = moveMountAlongBeam(selected, direction * .05);
    if (!changed) invalidFeedback('Эта сторона закреплена. Сначала освободи опору.');
  } else if (state.selection.kind === 'node') {
    selected.mass = clamp(selected.mass + direction, .5, 24);
    const weight = state.attachments.find((item) => item.nodeId === selected.id && item.type === 'weight');
    if (weight) weight.amount = selected.mass;
  } else if (state.selection.kind === 'attachment') {
    if (selected.type === 'motor') selected.amount = clamp(selected.amount + direction * .2, .4, 4);
    if (selected.type === 'weight') {
      selected.amount = clamp(selected.amount + direction * 2, 2, 30);
      const node = nodeById(selected.nodeId);
      if (node) node.mass = selected.amount;
    }
  } else if (state.selection.kind === 'constraint' && selected.type === 'spring') {
    selected.length = clamp(selected.length + direction * 12, 24, 500);
  } else {
    changed = false;
  }
  if (!changed && state.undo.length) state.undo.pop();
  updateInspector();
  saveSoon();
  clickSound(direction > 0 ? 350 : 240, .03);
}

function removeSelection() {
  if (!state.selection) return;
  pushUndo();
  removeThing(state.selection);
  state.selection = null;
  updateInspector();
  saveSoon();
  snapSound();
}

function removeThing(selection) {
  if (!selection) return;
  if (selection.kind === 'node') {
    const id = selection.id;
    state.nodes = state.nodes.filter((node) => node.id !== id);
    state.constraints = state.constraints.filter((constraint) => constraint.a !== id && constraint.b !== id);
    state.rotors = state.rotors.filter((rotor) => rotor.nodeId !== id);
    state.attachments = state.attachments.filter((item) => item.nodeId !== id && item.hostId !== id);
    state.joints = state.joints.filter((joint) => joint.a !== id && joint.b !== id);
  } else if (selection.kind === 'constraint') {
    const target = constraintById(selection.id);
    if (target?.type === 'beam') {
      const mounts = state.nodes.filter((node) => node.mount?.beamId === target.id).map((node) => node.id);
      for (const nodeId of mounts) removeThing({ kind: 'node', id: nodeId });
      state.joints = state.joints.filter((joint) => joint.beamA !== target.id && joint.beamB !== target.id);
    }
    state.constraints = state.constraints.filter((item) => item.id !== selection.id);
  } else if (selection.kind === 'rotor') {
    state.rotors = state.rotors.filter((item) => item.id !== selection.id);
  } else if (selection.kind === 'joint') {
    state.joints = state.joints.filter((item) => item.id !== selection.id);
  } else if (selection.kind === 'attachment') {
    const target = attachmentById(selection.id);
    if (target?.type === 'weight') {
      const node = nodeById(target.nodeId);
      if (node) node.mass = 1;
    }
    if (target?.type === 'payload') {
      removeThing({ kind: 'node', id: target.nodeId });
      return;
    }
    state.attachments = state.attachments.filter((item) => item.id !== selection.id);
  }
}

function resolveOrCreateNode(world) {
  return connectionPointAt(world, { create: true })?.node || createNode(world.x, world.y);
}
