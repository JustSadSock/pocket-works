function snap(value, step = GRID / 2) {
  return Math.round(value / step) * step;
}

function clamp(value, min, max) {
  return Math.max(min, Math.min(max, value));
}

function distance(a, b) {
  return Math.hypot(b.x - a.x, b.y - a.y);
}

function nodeById(id) {
  return state.nodes.find((node) => node.id === id) || null;
}

function rotorById(id) {

}

function attachmentById(id) {

function constraintById(id) {

function jointById(id) {

function beamById(id) {
  const beam = constraintById(id);
  return beam?.type === 'beam' ? beam : null;
}

function normalizeAngle(value) {
  let angle = value;
  while (angle > Math.PI) angle -= TAU;
  while (angle < -Math.PI) angle += TAU;
  return angle;
}

function itemLayer(item) {
  return Number.isFinite(item?.layer) ? item.layer : 0;
}

}

function projectPointToBeam(world, beam) {
  const a = nodeById(beam?.a);
  const b = nodeById(beam?.b);
  if (!a || !b) return null;
  const dx = b.x - a.x;
  const dy = b.y - a.y;
  const lengthSquared = dx * dx + dy * dy;
  const t = lengthSquared ? clamp(((world.x - a.x) * dx + (world.y - a.y) * dy) / lengthSquared, 0, 1) : 0;
  const point = { x: a.x + dx * t, y: a.y + dy * t };
  return { t, point, distance: Math.hypot(world.x - point.x, world.y - point.y) };
}

function syncMountNode(node, exact = false) {
  if (!node?.mount) return false;
  const beam = beamById(node.mount.beamId);
  const projection = projectPointToBeam(node, beam);
  if (!beam || !projection) return false;
  const a = nodeById(beam.a);
  const b = nodeById(beam.b);
  const t = clamp(node.mount.t, 0, 1);
  const x = a.x + (b.x - a.x) * t;
  const y = a.y + (b.y - a.y) * t;
  if (exact) {
    node.x = x;
    node.y = y;
    node.px = x;
    node.py = y;
  }
  return true;
}

function syncAllMounts(exact = false) {
  for (const node of state.nodes) syncMountNode(node, exact);
}

function mountNodeOnBeam(beam, world, options = {}) {
  const projection = projectPointToBeam(world, beam);
  if (!projection) return null;
  const a = nodeById(beam.a);
  const b = nodeById(beam.b);
  const endpointThreshold = Math.min(.08, 14 / Math.max(40, distance(a, b)));
  if (projection.t <= endpointThreshold) return a;
  if (projection.t >= 1 - endpointThreshold) return b;
  const existing = state.nodes.find((node) => node.mount?.beamId === beam.id && Math.abs(node.mount.t - projection.t) < .025);
  if (existing) return existing;
  return createNode(projection.point.x, projection.point.y, {
    role: 'mount',
    mass: .75,
    radius: 5,
    layer: options.layer ?? itemLayer(beam) + 1,
    mount: { beamId: beam.id, t: projection.t }
  });
}

function beamHitsAt(world, radius = 16 / state.camera.zoom, excludeId = null) {
  return state.constraints
    .filter((constraint) => constraint.type === 'beam' && constraint.id !== excludeId)
    .map((beam) => ({ beam, projection: projectPointToBeam(world, beam) }))
    .filter((hit) => hit.projection && hit.projection.distance <= radius)
    .sort((left, right) => itemLayer(right.beam) - itemLayer(left.beam) || left.projection.distance - right.projection.distance);
}

function connectionPointAt(world, { create = true, excludeBeamId = null, preferBeam = false } = {}) {
  const nearbyNode = findNodeAt(world, 16 / state.camera.zoom, { includeMounts: true });
  const beamHit = beamHitsAt(world, 15 / state.camera.zoom, excludeBeamId)[0] || null;
  if (nearbyNode && !preferBeam) {
    const beam = state.constraints.find((constraint) => constraint.type === 'beam' && constraint.id !== excludeBeamId && (constraint.a === nearbyNode.id || constraint.b === nearbyNode.id));
    return { node: nearbyNode, beamId: beam?.id || nearbyNode.mount?.beamId || null };
  }
  if (beamHit) {
    const node = create ? mountNodeOnBeam(beamHit.beam, beamHit.projection.point) : null;
    return node ? { node, beamId: beamHit.beam.id } : null;
  }
  if (nearbyNode) {
    const beam = state.constraints.find((constraint) => constraint.type === 'beam' && constraint.id !== excludeBeamId && (constraint.a === nearbyNode.id || constraint.b === nearbyNode.id));
    return { node: nearbyNode, beamId: beam?.id || nearbyNode.mount?.beamId || null };
  }
  if (!create) return null;
  const node = createNode(world.x, world.y);
  return { node, beamId: null };
}

function createJoint(type, first, second) {
  if (!first?.node || !second?.node || !first.beamId || !second.beamId || first.beamId === second.beamId) return null;
  const a = first.node;
  const b = second.node;
  const gap = distance(a, b);
  if (gap > 28) {
    if (!b.fixed && !b.mount) {
      b.x = a.x;
      b.y = a.y;
      b.px = a.x;
      b.py = a.y;
    } else {
      return null;
    }
  }
  const beamA = beamById(first.beamId);
  const beamB = beamById(second.beamId);
  if (!beamA || !beamB) return null;
  const angleA = Math.atan2(nodeById(beamA.b).y - nodeById(beamA.a).y, nodeById(beamA.b).x - nodeById(beamA.a).x);
  const angleB = Math.atan2(nodeById(beamB.b).y - nodeById(beamB.a).y, nodeById(beamB.b).x - nodeById(beamB.a).x);
  const existing = state.joints.find((joint) => joint.beamA === beamA.id && joint.beamB === beamB.id && joint.a === a.id && joint.b === b.id)
    || state.joints.find((joint) => joint.beamA === beamB.id && joint.beamB === beamA.id && joint.a === b.id && joint.b === a.id);
  if (existing) {
    existing.type = type;
    existing.targetAngle = normalizeAngle(angleB - angleA);
    return existing;
  }
  const joint = {
    id: nextId(type),
    type,
    a: a.id,
    b: b.id,
    beamA: beamA.id,
    beamB: beamB.id,
    targetAngle: normalizeAngle(angleB - angleA),
    stiffness: type === 'weld' ? .34 : 1,
    layer: Math.max(itemLayer(beamA), itemLayer(beamB), itemLayer(a), itemLayer(b)) + 1
  };
  state.joints.push(joint);
  return joint;
}

function createNode(x, y, options = {}) {
  const px = options.snap === false ? x : snap(x);
  const py = options.snap === false ? y : snap(y);
  const node = {
    id: nextId('node'),
    x: px,
    y: py,
    px,
    py,
    fixed: Boolean(options.fixed),
    mass: options.mass || 1,
    radius: options.radius || 6,
    role: options.role || 'node',
    layer: options.layer ?? 0,
    mount: options.mount ? { beamId: options.mount.beamId, t: clamp(options.mount.t, 0, 1) } : null
  };
  state.nodes.push(node);
  return node;
}

function createConstraint(type, aId, bId, options = {}) {
  const a = nodeById(aId);
  const b = nodeById(bId);
  if (!a || !b || aId === bId) return null;
  const existing = state.constraints.find((item) => item.a === aId && item.b === bId && item.type === type)
    || state.constraints.find((item) => item.a === bId && item.b === aId && item.type === type);
  if (existing) return existing;
  const natural = distance(a, b);
  const constraint = {
    id: nextId(type),
    type,
    a: aId,
    b: bId,
    length: options.length ?? natural,
    maxLength: options.maxLength ?? natural,
    stiffness: options.stiffness ?? (type === 'spring' ? 0.11 : 1),
    active: options.active ?? true,
    width: options.width ?? (type === 'beam' ? 10 : 3),
    layer: options.layer ?? Math.max(itemLayer(a), itemLayer(b))
  };
  state.constraints.push(constraint);
  return constraint;
}

function createRotor(type, nodeId, options = {}) {
  const node = nodeById(nodeId);
  if (!node) return null;
  const rotor = {
    id: nextId(type),
    type,
    nodeId,
    radius: options.radius ?? ({ wheel: 42, gear: 30, winch: 24, pulley: 20 }[type] || 28),
    angle: options.angle || 0,
    omega: options.omega || 0,
    teeth: options.teeth || Math.max(8, Math.round((options.radius || 30) / 3)),
    layer: options.layer ?? itemLayer(node) + 1
  };
  state.rotors.push(rotor);
  return rotor;
}

function createAttachment(type, nodeId, options = {}) {
  const node = nodeById(nodeId);
  if (!node) return null;
  const attachment = {
    id: nextId(type),
    type,
    nodeId,
    amount: options.amount ?? (type === 'weight' ? 10 : type === 'motor' ? 1.6 : 1),
    hostId: options.hostId || null,
    layer: options.layer ?? itemLayer(node) + 1
  };
  state.attachments.push(attachment);
  if (type === 'weight') node.mass = Math.max(node.mass, attachment.amount);
  if (type === 'payload') {
    node.role = 'payload';
    node.radius = 14;
    node.mass = Math.max(node.mass, 3.5);
  }
  return attachment;
}

function addBeam(a, b) {

function buildCart() {
  clearWorld();
  const left = createNode(340, 520);
  const right = createNode(560, 520);
  const topLeft = createNode(375, 455);
  const topRight = createNode(525, 455);
  addBeam(left, right);
  addBeam(left, topLeft);
  addBeam(topLeft, topRight);
  addBeam(topRight, right);
  addBeam(topLeft, right);
  addBeam(left, topRight);
  createRotor('wheel', left.id, { radius: 48 });
  createRotor('wheel', right.id, { radius: 48 });
  createRotor('gear', left.id, { radius: 28 });
  createRotor('gear', right.id, { radius: 28 });
  createAttachment('motor', left.id, { amount: 1.8 });
  createAttachment('motor', right.id, { amount: 1.8 });
  const ballast = createNode(450, 440, { mass: 6 });
  addBeam(topLeft, ballast);
  addBeam(ballast, topRight);
  createAttachment('weight', ballast.id, { amount: 6 });
  state.camera = { x: 450, y: 390, zoom: 1.03 };
}

function buildTrebuchet() {
  clearWorld();
  const baseL = createNode(300, 560);
  const baseR = createNode(650, 560);
  const pivot = createNode(470, 425);
  const footL = createNode(400, 560);
  const footR = createNode(540, 560);
  addBeam(baseL, baseR);
  addBeam(footL, pivot);
  addBeam(footR, pivot);
  addBeam(baseL, pivot);
  addBeam(baseR, pivot);
  const weight = createNode(350, 330, { mass: 14, radius: 11 });
  const armTip = createNode(690, 555);
  addBeam(weight, pivot);
  addBeam(pivot, armTip);
  createAttachment('weight', weight.id, { amount: 14 });
  const latch = createConstraint('latch', armTip.id, baseR.id, { stiffness: 1, active: true });
  latch.length = distance(armTip, baseR);
  const projectile = createNode(714, 534, { mass: 3.5, radius: 14, role: 'payload' });
  createAttachment('payload', projectile.id, { hostId: armTip.id });
  createConstraint('payloadLock', armTip.id, projectile.id, { stiffness: 1, active: true, length: distance(armTip, projectile) });
  createRotor('wheel', baseL.id, { radius: 34 });
  createRotor('wheel', baseR.id, { radius: 34 });
  state.camera = { x: 490, y: 390, zoom: .92 };
}

function serializeModel() {
  return {
    version: 2,
    idCounter,
    nodes: state.nodes,
    constraints: state.constraints,
    rotors: state.rotors,
    attachments: state.attachments,
    joints: state.joints,
    camera: state.camera
  };
}

function applyModel(model, { resetHistory = false } = {}) {
  if (!model || ![1, 2].includes(model.version) || !Array.isArray(model.nodes) || !Array.isArray(model.constraints)) throw new Error('Неподдерживаемый чертёж');
  state.nodes = deepClone(model.nodes).map((node) => ({ layer: 0, mount: null, ...node }));
  state.constraints = deepClone(model.constraints).map((constraint) => ({ layer: 0, ...constraint }));
  state.rotors = deepClone(model.rotors || []).map((rotor) => ({ layer: 1, ...rotor }));
  state.attachments = deepClone(model.attachments || []).map((attachment) => ({ layer: 1, ...attachment }));
  state.joints = deepClone(model.joints || []).map((joint) => ({ layer: 2, stiffness: joint.type === 'weld' ? .34 : 1, ...joint }));
  state.camera = { x: 480, y: 360, zoom: 1, ...(model.camera || {}) };
  idCounter = Math.max(Number(model.idCounter) || 1, inferNextId());
  state.selection = null;
  state.pendingNodeId = null;
  state.pendingJointTarget = null;
  state.hitCycle = null;
  state.mode = 'build';
  state.drive = 0;
  state.simTime = 0;
  fireCountdown = -1;
  if (resetHistory) {
    state.undo = [];
    state.redo = [];
  }
  syncAllMounts(true);
  resetVelocities();
  updateModeUI();
  updateInspector();
}

function inferNextId() {
  const ids = [...state.nodes, ...state.constraints, ...state.rotors, ...state.attachments, ...state.joints]
    .map((item) => Number(String(item.id).split('-').at(-1)))
    .filter(Number.isFinite);
  return ids.length ? Math.max(...ids) + 1 : 1;
}

function saveSoon() {
  state.dirty = true;
  clearTimeout(saveTimer);
  saveTimer = setTimeout(saveNow, 240);
}

function saveNow() {
  if (!state.dirty) return;
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(serializeModel()));
    state.dirty = false;
  } catch {
    showToast('Не удалось сохранить чертёж');
  }
}

function loadSaved() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return false;
    applyModel(JSON.parse(raw), { resetHistory: true });
    return true;
  } catch {
    localStorage.removeItem(STORAGE_KEY);
    return false;
  }
}

function snapshot() {
  return JSON.stringify(serializeModel());
}

function pushUndo() {
  state.undo.push(snapshot());
  if (state.undo.length > 50) state.undo.shift();
  state.redo = [];
  updateHistoryButtons();
}

function undo() {
  if (!state.undo.length || state.mode !== 'build') return;
  state.redo.push(snapshot());
  const model = JSON.parse(state.undo.pop());
  applyModel(model);
  saveSoon();
  updateHistoryButtons();
  clickSound(260, .035);
}
