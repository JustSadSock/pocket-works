function allToolDefinitionsV22() {
  const seen = new Set();
  const list = [];
  for (const [category, tools] of Object.entries(TOOL_GROUPS)) {
    for (const definition of tools) {
      if (seen.has(definition[0])) continue;
      seen.add(definition[0]);
      list.push({ category, definition });
    }
  }
  return list;
}

function toolDefinitionV22(tool) {
  return allToolDefinitionsV22().find((item) => item.definition[0] === tool) || null;
}

function matchingToolsV22() {
  const all = allToolDefinitionsV22();
  const query = (partsSearchV22?.value || '').trim().toLocaleLowerCase('ru');
  if (query) {
    return all.filter(({ definition }) => {
      const [tool, label] = definition;
      return `${label} ${TOOL_PURPOSES_V22[tool] || ''}`.toLocaleLowerCase('ru').includes(query);
    });
  }
  if (uxPrefsV22.filter === 'favourites') {
    return uxPrefsV22.favourites.map(toolDefinitionV22).filter(Boolean);
  }
  const categoryTools = all.filter((item) => item.category === state.category);
  const recent = uxPrefsV22.recent
    .map(toolDefinitionV22)
    .filter((item) => item && item.category === state.category);
  const recentIds = new Set(recent.map((item) => item.definition[0]));
  return [...recent, ...categoryTools.filter((item) => !recentIds.has(item.definition[0]))];
}

function contextKeyV22() {
  if (!state.selection) return null;
  const selected = getSelected();
  if (!selected) return null;
  if (state.selection.kind === 'constraint' && selected.type === 'beam') return 'beam';
  return state.selection.kind;
}

function showContextToolsV22() {
  installContextToolsV22();
  if (state.mode !== 'build' || state.tool !== 'select' || !state.selection) return hideContextToolsV22();
  const key = contextKeyV22();
  const tools = CONTEXT_TOOLS_V22[key] || [];
  const point = selectionWorld();
  if (!point || !tools.length) return hideContextToolsV22();
  contextToolsV22.innerHTML = `<div class="context-tools-title-v22">${key === 'rotor' ? 'НА ЭТОТ ВАЛ' : key === 'joint' ? 'ДЛЯ ЭТОГО ШАРНИРА' : key === 'beam' ? 'НА ЭТУ БАЛКУ' : 'ИЗ ЭТОГО УЗЛА'}</div>`;
  for (const tool of tools) {
    const item = toolDefinitionV22(tool);
    if (!item) continue;
    const button = document.createElement('button');
    button.type = 'button';
    button.dataset.nativePress = '';
    button.textContent = item.definition[1];
    button.addEventListener('click', (event) => {
      event.stopPropagation();
      const target = { ...point };
      setCategory(item.category);
      setTool(tool);
      requestAnimationFrame(() => handleBuildTap(target));
    });
    contextToolsV22.append(button);

  positionContextToolsV22();

}

const TWO_POINT_TOOLS_V22 = new Set(['beam', 'armor', 'hinge', 'limit', 'weld', 'rope', 'spring', 'damper', 'wing', 'canopy', 'bow', 'actuator']);

function activeTargetTypeV22() {
  if (TWO_POINT_TOOLS_V22.has(state.tool)) return 'node';
  if (NODE_TOOLS_V22.has(state.tool)) return 'node';
  if (SHAFT_TOOLS_V22.has(state.tool) || TRANSMISSION_TOOLS_V22.has(state.tool)) return 'shaft';
  if (state.tool === 'jointServo') return 'joint';
  if (state.tool === 'follower') return pendingCamRotorIdV20 ? 'node' : 'cam';
  if (state.tool === 'sequencer') return 'drum';
  return null;
}

function drawTargetRingV22(point, radius, label, strong = false) {
  ctx.save();
  ctx.strokeStyle = strong ? '#9b4b35' : '#41736b';
  ctx.fillStyle = strong ? 'rgba(155,75,53,.14)' : 'rgba(65,115,107,.12)';
  ctx.lineWidth = strong ? 3 : 2;
  ctx.setLineDash(strong ? [] : [4, 4]);
  ctx.beginPath();
  ctx.arc(point.x, point.y, radius, 0, TAU);
  ctx.fill();
  ctx.stroke();
  ctx.setLineDash([]);
  if (label) {
    ctx.font = '800 8px ui-monospace, monospace';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'bottom';
    ctx.fillStyle = strong ? '#843e2d' : '#315e58';
    ctx.fillText(label, point.x, point.y - radius - 3);
  }
  ctx.restore();
}

function drawCompatibleTargetsV22() {
  if (state.mode !== 'build') return;
  const targetType = activeTargetTypeV22();
  if (!targetType) return;
  const pulse = 1.5 + Math.sin(performance.now() / 180) * 1.5;
  if (targetType === 'node') {
    for (const node of state.nodes) {
      const point = worldToScreen(node.x, node.y);
      drawTargetRingV22(point, 10 + pulse, node.id === state.pendingNodeId ? 'ПЕРВАЯ' : '', node.id === state.pendingNodeId);
    }
  } else if (targetType === 'joint') {
    for (const joint of state.joints) {
      const a = nodeById(joint.a), b = nodeById(joint.b);
      if (a && b) drawTargetRingV22(worldToScreen((a.x + b.x) * .5, (a.y + b.y) * .5), 13 + pulse, 'ШАРНИР');
    }
  } else {
    const rotors = targetType === 'cam'
      ? state.rotors.filter((item) => item.type === 'cam')
      : targetType === 'drum'
        ? state.rotors.filter((item) => item.type === 'drum')
        : state.rotors;
    for (const rotor of rotors) {
      const node = nodeById(rotor.nodeId);
      if (node) drawTargetRingV22(worldToScreen(node.x, node.y), clamp(rotor.radius * state.camera.zoom + 8, 15, 54), targetType === 'shaft' ? 'ВАЛ' : targetType.toUpperCase());
    }
  }
  const pendingNode = state.pendingNodeId ? nodeById(state.pendingNodeId) : null;
  if (pendingNode && pointerWorldV22) {
    const a = worldToScreen(pendingNode.x, pendingNode.y);
    const b = worldToScreen(pointerWorldV22.x, pointerWorldV22.y);
    ctx.save();
    ctx.strokeStyle = '#a85a36';
    ctx.lineWidth = 3;
    ctx.setLineDash([8, 5]);
    ctx.beginPath();
    ctx.moveTo(a.x, a.y);
    ctx.lineTo(b.x, b.y);
    ctx.stroke();
    ctx.setLineDash([]);
    drawArrowHeadV21(b.x, b.y, Math.atan2(b.y - a.y, b.x - a.x), 8, '#a85a36');
    ctx.restore();
  }
}

function hasTransmissionV22() {
  if (state.constraints.some((item) => ['belt', 'chain', 'rack'].includes(item.type))) return true;
  return state.rotors.some((rotor) => rotor.type === 'gear' && closestMeshedGear(rotor));
}

function evaluateGuideV22() {
  const tests = {
    beam: state.constraints.some((item) => item.type === 'beam'),
    rotor: state.rotors.some((item) => ['wheel', 'gear'].includes(item.type)),
    power: state.attachments.some((item) => ['motor', 'mainspring'].includes(item.type)),
    transmission: hasTransmissionV22(),
    test: state.mode === 'run' && state.simTime >= 1.5
  };
  let changed = false;
  for (const [id, complete] of Object.entries(tests)) {
    if (complete && !guideProgressV22.has(id)) {
      guideProgressV22.add(id);
      changed = true;
    }
  }
  if (changed) {
    persistGuideV22();
    const count = guideProgressV22.size;
    clickSound(count >= GUIDE_TASKS_V22.length ? 620 : 460, count >= GUIDE_TASKS_V22.length ? .12 : .045);
    vibrate(count >= GUIDE_TASKS_V22.length ? [18, 35, 24] : 10);
  }
  renderGuideV22(changed);
}

  if (!force && signature === lastGuideSignatureV22) return;
  lastGuideSignatureV22 = signature;

    <div class="guide-head-v22"><span>ПУТЬ СИЛЫ · ${guideProgressV22.size}/${GUIDE_TASKS_V22.length}</span><button class="guide-close-v22" type="button" data-native-press aria-label="Скрыть обучение">×</button></div>

function installRunLabV22() {
  if (runLabV22) return;
  ui.runDock.classList.add('run-lab-enabled-v22');
  runLabV22 = document.createElement('div');
  runLabV22.className = 'run-lab-v22';
  runLabV22.innerHTML = `
    <button class="pause-v22" type="button" data-native-press aria-label="Пауза" aria-pressed="false">Ⅱ</button>
    <button class="step-v22" type="button" data-native-press>ШАГ</button>
    <button class="speed-v22" data-speed=".25" type="button" data-native-press>0,25×</button>
    <button class="speed-v22 active" data-speed="1" type="button" data-native-press>1×</button>
    <button class="speed-v22" data-speed="2" type="button" data-native-press>2×</button>
    <button class="reset-v22" type="button" data-native-press>↺ СНАЧАЛА</button>
  `;
  ui.runDock.prepend(runLabV22);
  runLabV22.querySelector('.pause-v22').addEventListener('click', () => {
    runPausedV22 = !runPausedV22;
    runAccumulatorV22 = 0;
    updateRunLabV22();
    clickSound(runPausedV22 ? 210 : 390, .035);
  });
  runLabV22.querySelector('.step-v22').addEventListener('click', () => {
    if (state.mode !== 'run') return;
    runPausedV22 = true;
    runAccumulatorV22 = 0;
    baseSimulateUxV22(16.6667);
    evaluateGuideV22();
    updateRunLabV22();
    clickSound(330, .02);
  });
  runLabV22.querySelectorAll('.speed-v22').forEach((button) => button.addEventListener('click', () => {
    runSpeedV22 = Number(button.dataset.speed) || 1;
    runPausedV22 = false;
    runAccumulatorV22 = 0;
    updateRunLabV22();
    clickSound(300 + runSpeedV22 * 80, .03);
  }));
  runLabV22.querySelector('.reset-v22').addEventListener('click', () => {
    if (state.mode !== 'run') return;
    returnToBuild();
    enterRunMode();
    runPausedV22 = false;
    updateRunLabV22();
    setYardNote('Испытание начато заново.', 1100);
  });
  updateRunLabV22();
}

function updateRunLabV22() {
  if (!runLabV22) return;
  const pause = runLabV22.querySelector('.pause-v22');
  pause.classList.toggle('active', runPausedV22);
  pause.textContent = runPausedV22 ? '▶' : 'Ⅱ';
  pause.setAttribute('aria-label', runPausedV22 ? 'Продолжить' : 'Пауза');
  pause.setAttribute('aria-pressed', runPausedV22 ? 'true' : 'false');
  runLabV22.querySelector('.step-v22').disabled = state.mode !== 'run';
  runLabV22.querySelectorAll('.speed-v22').forEach((button) => button.classList.toggle('active', Number(button.dataset.speed) === runSpeedV22 && !runPausedV22));
  ui.runDock.classList.toggle('paused-v22', runPausedV22);
}

const baseSimulateUxV22 = simulate;
simulate = function simulateWithLabV22(dt) {
  if (state.mode !== 'run' || runPausedV22) return;
  const quantum = 1000 / 60;
  runAccumulatorV22 += Math.min(50, dt) * runSpeedV22;
  let steps = 0;
  while (runAccumulatorV22 >= quantum && steps < 6) {
    baseSimulateUxV22(quantum);
    runAccumulatorV22 -= quantum;
    steps += 1;
  }
  if (steps === 6) runAccumulatorV22 = 0;
  evaluateGuideV22();
};

const baseEnterRunUxV22 = enterRunMode;
enterRunMode = function enterRunLabV22() {
  runPausedV22 = false;
  runAccumulatorV22 = 0;
  baseEnterRunUxV22();
  installRunLabV22();
  updateRunLabV22();
  hideContextToolsV22();
};

const baseReturnBuildUxV22 = returnToBuild;
returnToBuild = function returnFromRunLabV22() {
  baseReturnBuildUxV22();
  runPausedV22 = false;
  runAccumulatorV22 = 0;
  updateRunLabV22();
  renderGuideV22(true);
};

let visionLayersV22 = null;
const VISION_LAYERS_V22 = [
  ['all', 'ВСЁ'],
  ['motion', 'ДВИЖЕНИЕ'],
  ['energy', 'ЭНЕРГИЯ'],
  ['control', 'A / B'],
  ['forces', 'СИЛЫ'],
  ['check', 'ПРОВЕРКА']
];

function installVisionLayersV22() {
  if (visionLayersV22) return;
  installMechanicalVisionV21();
  visionLayersV22 = document.createElement('div');
  visionLayersV22.className = 'vision-layers-v22';
  for (const [value, label] of VISION_LAYERS_V22) {
    const button = document.createElement('button');
    button.type = 'button';
    button.dataset.visionLayer = value;
    button.dataset.nativePress = '';
    button.textContent = label;
    button.addEventListener('click', () => {
      state.visionLayerV22 = value;
      state.mechanicsVisionV21 = true;
      localStorage.setItem(VISION_LAYER_KEY_V22, value);
      localStorage.setItem(VISION_KEY_V21, 'on');
      updateMechanicalVisionUiV21();
      clickSound(360, .025);
    });
    visionLayersV22.append(button);
  }
  document.querySelector('.yard').append(visionLayersV22);
  updateVisionLayersV22();
}

function updateVisionLayersV22() {
  if (!visionLayersV22) return;
  visionLayersV22.hidden = !state.mechanicsVisionV21;
  visionLayersV22.querySelectorAll('button').forEach((button) => button.classList.toggle('active', button.dataset.visionLayer === state.visionLayerV22));
  document.querySelector('.yard').dataset.visionLayers = state.mechanicsVisionV21 ? 'on' : 'off';
}

  return links;
};
