    const projectile = createNode(node.x + 22, node.y - 18, { mass: 3.5, radius: 14, role: 'payload' });
    const payload = createAttachment('payload', projectile.id, { hostId: node.id });
    createConstraint('payloadLock', node.id, projectile.id, { stiffness: 1, active: true, length: distance(node, projectile) });
    selection = { kind: 'attachment', id: payload.id };
  }
  selectThing(selection);
  saveSoon();
  placedFeedback();
}

  clickSound(430, .035);
}

}

function beginPointer(event) {
  if (firstInteraction) {
    firstInteraction = false;
    ensureAudio();
  }
  const point = canvasPoint(event);
  const world = screenToWorld(point.x, point.y);
  canvas.setPointerCapture(event.pointerId);
  state.gesture = {
    pointerId: event.pointerId,
    startPoint: point,
    lastPoint: point,
    startWorld: world,
    moved: false,
    startCamera: { ...state.camera }
  };

  if (state.mode !== 'build') return;
  if (state.tool === 'select') {
    const node = findNodeAt(world, 22 / state.camera.zoom);
    if (node) {
      pushUndo();
      state.drag = { nodeId: node.id, startX: node.x, startY: node.y };
      selectThing({ kind: 'node', id: node.id });
      return;
    }
  }
  if (state.tool === 'pan') {
    state.drag = { pan: true };
  }
}

function movePointer(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;
  const point = canvasPoint(event);
  const dx = point.x - gesture.startPoint.x;
  const dy = point.y - gesture.startPoint.y;
  if (Math.hypot(dx, dy) > 5) gesture.moved = true;
  gesture.lastPoint = point;

  if (state.mode !== 'build' || !state.drag) return;
  if (state.drag.pan) {
    state.camera.x = gesture.startCamera.x - dx / state.camera.zoom;
    state.camera.y = gesture.startCamera.y - dy / state.camera.zoom;
    return;
  }
  const node = nodeById(state.drag.nodeId);
  if (!node) return;
  const world = screenToWorld(point.x, point.y);
  node.x = snap(world.x, GRID / 4);
  node.y = snap(world.y, GRID / 4);
  node.px = node.x;
  node.py = node.y;
  state.dirty = true;
}

function endPointer(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;
  try { canvas.releasePointerCapture(event.pointerId); } catch {}
  const point = canvasPoint(event);
  const world = screenToWorld(point.x, point.y);
  if (state.mode === 'build') {
    if (state.drag) {
      if (state.drag.pan) {
        if (!gesture.moved) setYardNote('Тяни поле, чтобы перемещать камеру.', 1000);
      } else if (gesture.moved) {
        saveSoon();
        placedFeedback();
      } else {
        if (state.undo.length) state.undo.pop();
        updateHistoryButtons();
      }
    } else if (!gesture.moved) {
      handleBuildTap(world);
    }
  }
  state.drag = null;
  state.gesture = null;
}

function cancelPointer(event) {
  if (!state.gesture || state.gesture.pointerId !== event.pointerId) return;
  if (state.drag?.nodeId) {
    const node = nodeById(state.drag.nodeId);
    if (node) {
      node.x = state.drag.startX;
      node.y = state.drag.startY;
      node.px = node.x;
      node.py = node.y;
    }
    if (state.undo.length) state.undo.pop();
  }
  state.drag = null;
  state.gesture = null;
  updateHistoryButtons();
}

function zoomBy(factor) {
  state.camera.zoom = clamp(state.camera.zoom * factor, .42, 2.4);
  saveSoon();
  clickSound(factor > 1 ? 410 : 260, .025);
}

function frameConstruction() {
  if (!state.nodes.length) {
    state.camera = { x: 480, y: 360, zoom: 1 };
    return;
  }
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
  for (const node of state.nodes) {
    let extra = node.radius || 8;
    for (const rotor of state.rotors) if (rotor.nodeId === node.id) extra = Math.max(extra, rotor.radius);
    minX = Math.min(minX, node.x - extra);
    maxX = Math.max(maxX, node.x + extra);
    minY = Math.min(minY, node.y - extra);
    maxY = Math.max(maxY, node.y + extra);
  }
  const contentWidth = Math.max(120, maxX - minX);
  const contentHeight = Math.max(120, maxY - minY);
  state.camera.x = (minX + maxX) / 2;
  state.camera.y = (minY + maxY) / 2;
  state.camera.zoom = clamp(Math.min((width - 70) / contentWidth, (height - 70) / contentHeight), .45, 1.55);
  saveSoon();
}

function simulate(dt) {
  if (state.mode !== 'run') return;
  const step = clamp(dt / 16.6667, .35, 1.7);
  state.simTime += dt / 1000;

  if (fireCountdown >= 0) {
    fireCountdown -= dt / 1000;
    if (fireCountdown <= 0) {
      fireCountdown = -1;
      releasePayloads();
    }
  }

  driveRotors(step);
  integrateNodes(step);
  for (let iteration = 0; iteration < 9; iteration += 1) {
    solveConstraints();
    solveWheelGround(step);
    solveNodeGround();
  }
  solveGearCoupling(step);
  updateRotorAngles(step);
  updateWinches(step);
  followRunCamera(step);
}

function followRunCamera(step) {
  const focusNodes = state.nodes.filter((node) => node.role !== 'payload');
  if (!focusNodes.length) return;
  const centreX = focusNodes.reduce((sum, node) => sum + node.x, 0) / focusNodes.length;
  const centreY = focusNodes.reduce((sum, node) => sum + node.y, 0) / focusNodes.length;
  const targetY = clamp(centreY - 80, 260, GROUND_Y - 70);
  const follow = 1 - Math.pow(.93, step);
  state.camera.x += (centreX - state.camera.x) * follow;
  state.camera.y += (targetY - state.camera.y) * follow;
}

function inverseMass(node) {
  return node.fixed ? 0 : 1 / Math.max(.2, node.mass || 1);
}

function integrateNodes(step) {
  const gravity = .52 * step * step;
  for (const node of state.nodes) {
    if (node.fixed) {
      node.px = node.x;
      node.py = node.y;
      continue;
    }
    const vx = (node.x - node.px) * .995;
    const vy = (node.y - node.py) * .995;
    node.px = node.x;
    node.py = node.y;
    node.x += vx;
    node.y += vy + gravity;
  }
}

function solveConstraints() {
  for (const constraint of state.constraints) {
    if (!constraint.active) continue;
    const a = nodeById(constraint.a);
    const b = nodeById(constraint.b);
    if (!a || !b) continue;
    const dx = b.x - a.x;
    const dy = b.y - a.y;
    const dist = Math.hypot(dx, dy) || .0001;
    const target = constraint.type === 'rope' ? constraint.maxLength : constraint.length;
    if (constraint.type === 'rope' && dist <= target) continue;
    const diff = (dist - target) / dist;
    const stiffness = constraint.stiffness ?? 1;
    const invA = inverseMass(a);
    const invB = inverseMass(b);
    const total = invA + invB;
    if (!total) continue;
    const correctionX = dx * diff * stiffness;
    const correctionY = dy * diff * stiffness;
    a.x += correctionX * (invA / total);
    a.y += correctionY * (invA / total);
    b.x -= correctionX * (invB / total);
    b.y -= correctionY * (invB / total);
  }
}

function solveNodeGround() {
  for (const node of state.nodes) {
    if (state.rotors.some((rotor) => rotor.nodeId === node.id && rotor.type === 'wheel')) continue;
    const radius = node.radius || 6;
    const floor = GROUND_Y - radius;
    if (node.y > floor) {
      node.y = floor;
      const vx = node.x - node.px;
      node.px = node.x - vx * .78;
      node.py = node.y + (node.y - node.py) * .16;
    }
  }
}

function shaftGroups() {
  const groups = new Map();
  for (const rotor of state.rotors) {
    if (!groups.has(rotor.nodeId)) groups.set(rotor.nodeId, []);
    groups.get(rotor.nodeId).push(rotor);
  }
  return groups;
}

function driveRotors(step) {
  const motors = state.attachments.filter((item) => item.type === 'motor');
  const groups = shaftGroups();
  for (const motor of motors) {
    const rotors = groups.get(motor.nodeId) || [];
    const target = state.drive * motor.amount * .055;
    for (const rotor of rotors) rotor.omega += (target - rotor.omega) * .22 * step;
  }
  for (const rotors of groups.values()) {
    if (rotors.length < 2) continue;
    const average = rotors.reduce((sum, rotor) => sum + rotor.omega, 0) / rotors.length;
    for (const rotor of rotors) rotor.omega += (average - rotor.omega) * .65;
  }
}

function solveGearCoupling(step) {
  const gears = state.rotors.filter((rotor) => rotor.type === 'gear');
  for (let i = 0; i < gears.length; i += 1) {
    const a = gears[i];
    const nodeA = nodeById(a.nodeId);
    if (!nodeA) continue;
    for (let j = i + 1; j < gears.length; j += 1) {
      const b = gears[j];
      if (a.nodeId === b.nodeId) continue;
      const nodeB = nodeById(b.nodeId);
      if (!nodeB) continue;
      const d = distance(nodeA, nodeB);
      const meshDistance = a.radius + b.radius;
      if (Math.abs(d - meshDistance) > 10) continue;
      const targetB = -a.omega * (a.radius / b.radius);
      const targetA = -b.omega * (b.radius / a.radius);
      b.omega += (targetB - b.omega) * .32 * step;
      a.omega += (targetA - a.omega) * .16 * step;
    }
  }
}

function solveWheelGround(step) {
  for (const wheel of state.rotors.filter((rotor) => rotor.type === 'wheel')) {
    const node = nodeById(wheel.nodeId);
    if (!node || node.fixed) continue;
    const floor = GROUND_Y - wheel.radius;
    if (node.y > floor) {
      node.y = floor;
      const currentVx = node.x - node.px;
      const surfaceVelocity = wheel.omega * wheel.radius * .75;
      const correction = clamp(surfaceVelocity - currentVx, -2.4, 2.4) * .3 * step;
      node.x += correction;
      node.px -= correction * .25;
      node.py = node.y + (node.y - node.py) * .12;
      if (Math.abs(state.drive) < .01) wheel.omega += ((currentVx / Math.max(10, wheel.radius)) - wheel.omega) * .3;
    }
  }
}

function updateRotorAngles(step) {
  for (const rotor of state.rotors) {
    rotor.omega *= rotor.type === 'wheel' ? .994 : .982;
    rotor.angle += rotor.omega * step;
  }
}

function updateWinches(step) {
  if (!activeWinchDirection) return;
  const winchNodes = new Set(state.rotors.filter((rotor) => rotor.type === 'winch').map((rotor) => rotor.nodeId));
  for (const rope of state.constraints.filter((item) => item.type === 'rope')) {
    if (!winchNodes.has(rope.a) && !winchNodes.has(rope.b)) continue;
    rope.maxLength = clamp(rope.maxLength - activeWinchDirection * 1.5 * step, 20, 700);
  }
}

function render() {
  resizeCanvas();
  ctx.save();
  ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  drawBackground();
  drawGround();
  drawConstraints();
  drawRotors();
  drawNodesAndAttachments();
  drawSelection();
  drawPending();
  ctx.restore();
}

function drawBackground() {
  ctx.fillStyle = '#e6d5ad';
  ctx.fillRect(0, 0, width, height);
  const zoom = state.camera.zoom;
  const major = GRID * zoom;
  if (major > 6) {
    const origin = worldToScreen(0, 0);
    ctx.lineWidth = 1;
    for (let x = ((origin.x % major) + major) % major; x < width; x += major) {
      const worldX = Math.round(screenToWorld(x, 0).x / GRID);
      ctx.strokeStyle = worldX % 5 === 0 ? 'rgba(41,78,96,.19)' : 'rgba(41,78,96,.08)';
      ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, height); ctx.stroke();
    }
    for (let y = ((origin.y % major) + major) % major; y < height; y += major) {
      const worldY = Math.round(screenToWorld(0, y).y / GRID);
      ctx.strokeStyle = worldY % 5 === 0 ? 'rgba(41,78,96,.19)' : 'rgba(41,78,96,.08)';
      ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(width, y); ctx.stroke();
    }
  }
  const wash = ctx.createLinearGradient(0, 0, 0, height);
  wash.addColorStop(0, 'rgba(255,249,224,.18)');
  wash.addColorStop(1, 'rgba(104,72,35,.1)');
  ctx.fillStyle = wash;
  ctx.fillRect(0, 0, width, height);
}

function drawGround() {
  const left = screenToWorld(0, 0).x - 60;
  const right = screenToWorld(width, 0).x + 60;
  const a = worldToScreen(left, GROUND_Y);
  const b = worldToScreen(right, GROUND_Y);
  if (a.y < -30 || a.y > height + 40) return;
  ctx.fillStyle = '#6f5a38';
  ctx.fillRect(0, a.y, width, height - a.y + 2);
  ctx.strokeStyle = '#3e3424';
  ctx.lineWidth = 4;
  ctx.beginPath(); ctx.moveTo(0, a.y); ctx.lineTo(width, b.y); ctx.stroke();
  ctx.strokeStyle = 'rgba(238,221,173,.22)';
  ctx.lineWidth = 1;
  for (let x = -30; x < width + 50; x += 54) {
    ctx.beginPath();
    ctx.moveTo(x, a.y + 12);
    ctx.lineTo(x + 42, a.y + 12);
    ctx.lineTo(x + 48, a.y + 29);
    ctx.lineTo(x + 6, a.y + 29);
    ctx.closePath();
    ctx.stroke();
  }
}

function drawConstraints() {
  for (const constraint of state.constraints) {
    if (!constraint.active && state.mode === 'run') continue;
    const aNode = nodeById(constraint.a);
    const bNode = nodeById(constraint.b);
    if (!aNode || !bNode) continue;
    const a = worldToScreen(aNode.x, aNode.y);
    const b = worldToScreen(bNode.x, bNode.y);
    if (constraint.type === 'beam') drawBeam(a, b, constraint.width * state.camera.zoom);
    else if (constraint.type === 'rope') drawRope(a, b, constraint, aNode, bNode);
    else if (constraint.type === 'spring') drawSpring(a, b, constraint);
    else if (constraint.type === 'latch') drawLatch(a, b);
    else if (constraint.type === 'payloadLock') drawPayloadLock(a, b);
  }
}

function drawBeam(a, b, thickness) {
  ctx.save();
  ctx.lineCap = 'round';
  ctx.strokeStyle = '#5e3d25';
  ctx.lineWidth = clamp(thickness + 4, 5, 19);
  ctx.beginPath(); ctx.moveTo(a.x, a.y); ctx.lineTo(b.x, b.y); ctx.stroke();
  ctx.strokeStyle = '#a47743';
  ctx.lineWidth = clamp(thickness, 3, 15);
  ctx.beginPath(); ctx.moveTo(a.x, a.y); ctx.lineTo(b.x, b.y); ctx.stroke();
  ctx.strokeStyle = 'rgba(245,213,149,.34)';
  ctx.lineWidth = Math.max(1, thickness * .16);
  ctx.beginPath(); ctx.moveTo(a.x, a.y - thickness * .12); ctx.lineTo(b.x, b.y - thickness * .12); ctx.stroke();
  ctx.restore();
}

function drawRope(a, b, constraint, aNode, bNode) {
  const current = distance(aNode, bNode);
  const slack = clamp((constraint.maxLength - current) * state.camera.zoom * .55, 0, 60);
  const controlX = (a.x + b.x) / 2;
  const controlY = (a.y + b.y) / 2 + slack;
  ctx.save();
  ctx.lineCap = 'round';
  ctx.strokeStyle = '#4a3824';
  ctx.lineWidth = 5;
  ctx.beginPath(); ctx.moveTo(a.x, a.y); ctx.quadraticCurveTo(controlX, controlY, b.x, b.y); ctx.stroke();
  ctx.strokeStyle = '#b89a61';
  ctx.lineWidth = 2;
  ctx.setLineDash([5, 4]);
  ctx.beginPath(); ctx.moveTo(a.x, a.y); ctx.quadraticCurveTo(controlX, controlY, b.x, b.y); ctx.stroke();
  ctx.restore();
}

function drawSpring(a, b, constraint) {
  const dx = b.x - a.x;
  const dy = b.y - a.y;
  const length = Math.hypot(dx, dy) || 1;
  const nx = -dy / length;
  const ny = dx / length;
  const coils = clamp(Math.round(length / 18), 5, 18);
  ctx.save();
  ctx.strokeStyle = '#687078';
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.moveTo(a.x, a.y);
  for (let i = 1; i < coils; i += 1) {
    const t = i / coils;
