function handleBuildTap(world) {
  if (state.tool === 'select') {
    cycleSelectionAt(world);
    return;
  }
  if (state.tool === 'layers') {
    cycleSelectionAt(world);
    return;
  }
  if (state.tool === 'erase') {
    const thing = findThingAt(world);
    if (!thing) return invalidFeedback('Здесь нечего стирать');
    pushUndo();
    removeThing(thing);
    saveSoon();
    snapSound();
    return;
  }
  if (state.tool === 'node') {
    pushUndo();
    const node = createNode(world.x, world.y);
    selectThing({ kind: 'node', id: node.id });
    saveSoon();
    placedFeedback();
    return;
  }
  if (['hinge', 'weld'].includes(state.tool)) {
    const target = connectionPointAt(world, {
      create: true,
      excludeBeamId: state.pendingJointTarget?.beamId || null,
      preferBeam: true
    });
    if (!target?.beamId) return invalidFeedback('Нужно коснуться балки');
    if (!state.pendingJointTarget) {
      pushUndo();
      state.pendingJointTarget = { nodeId: target.node.id, beamId: target.beamId };
      setYardNote('Первая балка выбрана. Укажи вторую в этой же точке.');
      clickSound(320, .025);
      return;
    }
    const first = { node: nodeById(state.pendingJointTarget.nodeId), beamId: state.pendingJointTarget.beamId };
    const joint = createJoint(state.tool, first, target);
    state.pendingJointTarget = null;
    if (!joint) {
      if (state.undo.length) state.undo.pop();
      updateHistoryButtons();
      return invalidFeedback('Балки должны пересекаться или касаться в точке крепления');
    }
    selectThing({ kind: 'joint', id: joint.id });
    saveSoon();
    placedFeedback();
    setYardNote(state.tool === 'weld' ? 'Скоба установлена: угол зафиксирован.' : 'Подвижный шарнир установлен.', 1700);
    return;
  }
  if (['beam', 'rope', 'spring', 'latch'].includes(state.tool)) {
    if (!state.pendingNodeId) pushUndo();
    const target = connectionPointAt(world, { create: true });
    const node = target?.node;
    if (!node) return invalidFeedback('Не удалось создать точку крепления');
    if (!state.pendingNodeId) {
      state.pendingNodeId = node.id;
      saveSoon();
      setYardNote('Первая точка выбрана. Укажи вторую.');
      clickSound(320, .025);
      return;
    }
    if (state.pendingNodeId === node.id) return invalidFeedback('Нужна другая точка');
    const type = state.tool;
    const constraint = createConstraint(type, state.pendingNodeId, node.id, {
      stiffness: type === 'spring' ? .11 : 1,
      width: type === 'beam' ? 10 : 3
    });
    state.pendingNodeId = null;
    if (constraint) {
      selectThing({ kind: 'constraint', id: constraint.id });
      saveSoon();
      placedFeedback();
      setYardNote(type === 'latch' ? 'Защёлка готова. Она откроется по «Спуску».' : 'Связь установлена.', 1500);
    }
    return;
  }

  pushUndo();
  const node = resolveOrCreateNode(world);
  let selection = { kind: 'node', id: node.id };
  if (['wheel', 'gear', 'winch', 'pulley'].includes(state.tool)) {
    const rotor = createRotor(state.tool, node.id);
    selection = { kind: 'rotor', id: rotor.id };
  } else if (state.tool === 'motor') {
    const existingRotor = state.rotors.some((rotor) => rotor.nodeId === node.id);
    if (!existingRotor) createRotor('gear', node.id, { radius: 24 });
    const motor = createAttachment('motor', node.id, { amount: 1.6 });
    selection = { kind: 'attachment', id: motor.id };
  } else if (state.tool === 'weight') {
    const existing = state.attachments.find((item) => item.nodeId === node.id && item.type === 'weight');
    if (existing) {
      existing.amount = clamp(existing.amount + 4, 2, 30);
      node.mass = existing.amount;
      selection = { kind: 'attachment', id: existing.id };
    } else {
      const weight = createAttachment('weight', node.id, { amount: 10 });
      selection = { kind: 'attachment', id: weight.id };
    }
  } else if (state.tool === 'anchor') {
    node.fixed = !node.fixed;
    node.px = node.x;
    node.py = node.y;
    selection = { kind: 'node', id: node.id };
  } else if (state.tool === 'payload') {
    const projectile = createNode(node.x + 22, node.y - 18, { mass: 3.5, radius: 14, role: 'payload', layer: itemLayer(node) + 1 });
    const payload = createAttachment('payload', projectile.id, { hostId: node.id });
    createConstraint('payloadLock', node.id, projectile.id, { stiffness: 1, active: true, length: distance(node, projectile) });
    selection = { kind: 'attachment', id: payload.id };
  }
  selectThing(selection);
  saveSoon();
  placedFeedback();
}

function placedFeedback() {
  vibrate(8);
  clickSound(430, .035);
}

function invalidFeedback(message) {
  setYardNote(message, 1200);
  vibrate([8, 24, 8]);
  clickSound(120, .05);
}

function beginPointer(event) {
  if (firstInteraction) {
    firstInteraction = false;
    ensureAudio();
  }
  const point = canvasPoint(event);
  const world = screenToWorld(point.x, point.y);
  canvas.setPointerCapture(event.pointerId);
  state.gesture = {
    pointerId: event.pointerId,
    startPoint: point,
    lastPoint: point,
    startWorld: world,
    moved: false,
    startCamera: { ...state.camera }
  };

  if (state.mode !== 'build') return;
  if (state.tool === 'select') {
    const node = findNodeAt(world, 22 / state.camera.zoom);
    if (node) {
      pushUndo();
      state.drag = { nodeId: node.id, startX: node.x, startY: node.y, startT: node.mount?.t ?? null };
      selectThing({ kind: 'node', id: node.id });
      return;
    }
  }
  if (state.tool === 'layers' && state.selection?.kind === 'node') {
    const selected = nodeById(state.selection.id);
    if (selected?.mount && distance(selected, world) < 24 / state.camera.zoom) {
      pushUndo();
      state.drag = { nodeId: selected.id, startX: selected.x, startY: selected.y, startT: selected.mount.t };
      return;
    }
  }
  if (state.tool === 'pan') state.drag = { pan: true };
}

function movePointer(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;
  const point = canvasPoint(event);
  const dx = point.x - gesture.startPoint.x;
  const dy = point.y - gesture.startPoint.y;
  if (Math.hypot(dx, dy) > 5) gesture.moved = true;
  gesture.lastPoint = point;

  if (state.mode !== 'build' || !state.drag) return;
  if (state.drag.pan) {
    state.camera.x = gesture.startCamera.x - dx / state.camera.zoom;
    state.camera.y = gesture.startCamera.y - dy / state.camera.zoom;
    return;
  }
  const node = nodeById(state.drag.nodeId);
  if (!node) return;
  const world = screenToWorld(point.x, point.y);
  if (node.mount) {
    const beam = beamById(node.mount.beamId);
    const projection = projectPointToBeam(world, beam);
    if (projection) {
      const deltaT = projection.t - node.mount.t;
      moveMountAlongBeam(node, deltaT);
    }
  } else {
    node.x = snap(world.x, GRID / 4);
    node.y = snap(world.y, GRID / 4);
    node.px = node.x;
    node.py = node.y;
    syncAllMounts(true);
  }
  state.dirty = true;
}

function endPointer(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;
  try { canvas.releasePointerCapture(event.pointerId); } catch {}
  const point = canvasPoint(event);
  const world = screenToWorld(point.x, point.y);
  if (state.mode === 'build') {
    if (state.drag) {
      if (state.drag.pan) {
        if (!gesture.moved) setYardNote('Тяни поле, чтобы перемещать камеру.', 1000);
      } else if (gesture.moved) {
        saveSoon();
        placedFeedback();
      } else {
        if (state.undo.length) state.undo.pop();
        updateHistoryButtons();
      }
    } else if (!gesture.moved) {
      handleBuildTap(world);
    }
  }
  state.drag = null;
  state.gesture = null;
}

function cancelPointer(event) {
  if (!state.gesture || state.gesture.pointerId !== event.pointerId) return;
  if (state.drag?.nodeId) {
    const node = nodeById(state.drag.nodeId);
    if (node) {
      node.x = state.drag.startX;
      node.y = state.drag.startY;
      if (node.mount && state.drag.startT !== null) node.mount.t = state.drag.startT;
      node.px = node.x;
      node.py = node.y;
      syncAllMounts(true);
    }
    if (state.undo.length) state.undo.pop();
  }
  state.drag = null;
  state.gesture = null;
  updateHistoryButtons();
}

function zoomBy(factor) {
  state.camera.zoom = clamp(state.camera.zoom * factor, .42, 2.4);
  saveSoon();
  clickSound(factor > 1 ? 410 : 260, .025);
}

function frameConstruction() {
  if (!state.nodes.length) {
    state.camera = { x: 480, y: 360, zoom: 1 };
    return;
  }
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
  for (const node of state.nodes) {
    let extra = node.radius || 8;
    for (const rotor of state.rotors) if (rotor.nodeId === node.id) extra = Math.max(extra, rotor.radius);
    minX = Math.min(minX, node.x - extra);
    maxX = Math.max(maxX, node.x + extra);
    minY = Math.min(minY, node.y - extra);
    maxY = Math.max(maxY, node.y + extra);
  }
  const contentWidth = Math.max(120, maxX - minX);
  const contentHeight = Math.max(120, maxY - minY);
  state.camera.x = (minX + maxX) / 2;
  state.camera.y = (minY + maxY) / 2;
  state.camera.zoom = clamp(Math.min((width - 70) / contentWidth, (height - 70) / contentHeight), .45, 1.55);
  saveSoon();
}

function simulate(dt) {
  if (state.mode !== 'run') return;
  const step = clamp(dt / 16.6667, .35, 1.7);
  state.simTime += dt / 1000;

  if (fireCountdown >= 0) {
    fireCountdown -= dt / 1000;
    if (fireCountdown <= 0) {
      fireCountdown = -1;
      releasePayloads();
    }
  }

  driveRotors(step);
  integrateNodes(step);
  for (let iteration = 0; iteration < 11; iteration += 1) {
    solveConstraints();
    solveMounts();
    solveJoints();
    solveFixedJoints();
    solveWheelGround(step);
    solveNodeGround();
  }
  solveGearCoupling(step);
  updateRotorAngles(step);
  updateWinches(step);
  followRunCamera(step);
}

function followRunCamera(step) {
  const focusNodes = state.nodes.filter((node) => node.role !== 'payload');
  if (!focusNodes.length) return;
  const centreX = focusNodes.reduce((sum, node) => sum + node.x, 0) / focusNodes.length;
  const centreY = focusNodes.reduce((sum, node) => sum + node.y, 0) / focusNodes.length;
  const targetY = clamp(centreY - 80, 260, GROUND_Y - 70);
  const follow = 1 - Math.pow(.93, step);
  state.camera.x += (centreX - state.camera.x) * follow;
  state.camera.y += (targetY - state.camera.y) * follow;
}

function inverseMass(node) {
  return node.fixed ? 0 : 1 / Math.max(.2, node.mass || 1);
}

function integrateNodes(step) {
  const gravity = .52 * step * step;
  for (const node of state.nodes) {
    if (node.fixed) {
      node.px = node.x;
      node.py = node.y;
      continue;
    }
    const vx = (node.x - node.px) * .995;
    const vy = (node.y - node.py) * .995;
    node.px = node.x;
    node.py = node.y;
    node.x += vx;
    node.y += vy + gravity;
  }
}

function solveConstraints() {
  for (const constraint of state.constraints) {
    if (!constraint.active) continue;
    const a = nodeById(constraint.a);
    const b = nodeById(constraint.b);
    if (!a || !b) continue;
    const dx = b.x - a.x;
    const dy = b.y - a.y;
    const dist = Math.hypot(dx, dy) || .0001;
    const target = constraint.type === 'rope' ? constraint.maxLength : constraint.length;
    if (constraint.type === 'rope' && dist <= target) continue;
    const diff = (dist - target) / dist;
    const stiffness = constraint.stiffness ?? 1;
    const invA = inverseMass(a);
    const invB = inverseMass(b);
    const total = invA + invB;
    if (!total) continue;
    const correctionX = dx * diff * stiffness;
    const correctionY = dy * diff * stiffness;
    a.x += correctionX * (invA / total);
    a.y += correctionY * (invA / total);
    b.x -= correctionX * (invB / total);
    b.y -= correctionY * (invB / total);
  }
}

function solveMounts() {
  for (const node of state.nodes) {
    if (!node.mount) continue;
    const beam = beamById(node.mount.beamId);
    const a = nodeById(beam?.a);
    const b = nodeById(beam?.b);
    if (!beam || !a || !b) continue;
    const t = clamp(node.mount.t, 0, 1);
    const targetX = a.x + (b.x - a.x) * t;
    const targetY = a.y + (b.y - a.y) * t;
    const errorX = node.x - targetX;
    const errorY = node.y - targetY;
    const invN = inverseMass(node);
    const invA = inverseMass(a);
    const invB = inverseMass(b);
    const wa = 1 - t;
    const wb = t;
    const total = invN + invA * wa * wa + invB * wb * wb;
    if (!total) continue;
    node.x -= errorX * invN / total;
    node.y -= errorY * invN / total;
    a.x += errorX * invA * wa / total;
    a.y += errorY * invA * wa / total;
    b.x += errorX * invB * wb / total;
    b.y += errorY * invB * wb / total;
  }
}

function solveJointPin(joint) {
  const a = nodeById(joint.a);
  const b = nodeById(joint.b);
  if (!a || !b || a.id === b.id) return;
  const dx = b.x - a.x;
  const dy = b.y - a.y;
  const invA = inverseMass(a);
  const invB = inverseMass(b);
  const total = invA + invB;
  if (!total) return;
  a.x += dx * invA / total;
  a.y += dy * invA / total;
  b.x -= dx * invB / total;
  b.y -= dy * invB / total;
}

function solveJoints() {
  for (const joint of state.joints) solveJointPin(joint);
}

function rotateBeamAround(beam, pivot, delta) {
  const endpoints = [nodeById(beam?.a), nodeById(beam?.b)];
  if (endpoints.some((node) => !node || node.fixed)) return false;
  const movable = endpoints.filter((node) => !node.mount);
  if (!movable.length) return false;
  const cos = Math.cos(delta);
  const sin = Math.sin(delta);
  for (const node of movable) {
    const x = node.x - pivot.x;
    const y = node.y - pivot.y;
    node.x = pivot.x + x * cos - y * sin;
    node.y = pivot.y + x * sin + y * cos;
  }
  return true;
}

function solveFixedJoints() {
  for (const joint of state.joints) {
    if (joint.type !== 'weld') continue;
    const beamA = beamById(joint.beamA);
    const beamB = beamById(joint.beamB);
    const nodeA = nodeById(joint.a);
    const nodeB = nodeById(joint.b);
    if (!beamA || !beamB || !nodeA || !nodeB) continue;
    const beamANodeA = nodeById(beamA.a);
    const beamANodeB = nodeById(beamA.b);
    const beamBNodeA = nodeById(beamB.a);
    const beamBNodeB = nodeById(beamB.b);
    const angleA = Math.atan2(beamANodeB.y - beamANodeA.y, beamANodeB.x - beamANodeA.x);
    const angleB = Math.atan2(beamBNodeB.y - beamBNodeA.y, beamBNodeB.x - beamBNodeA.x);
    const error = normalizeAngle((angleB - angleA) - joint.targetAngle);
    if (Math.abs(error) < .0005) continue;
    const pivot = { x: (nodeA.x + nodeB.x) / 2, y: (nodeA.y + nodeB.y) / 2 };
    const canA = !beamANodeA.fixed && !beamANodeB.fixed && [beamANodeA, beamANodeB].some((node) => !node.mount);
    const canB = !beamBNodeA.fixed && !beamBNodeB.fixed && [beamBNodeA, beamBNodeB].some((node) => !node.mount);
    const total = Number(canA) + Number(canB);
    if (!total) continue;
    const strength = joint.stiffness ?? .34;
    if (canA) rotateBeamAround(beamA, pivot, error * strength / total);
    if (canB) rotateBeamAround(beamB, pivot, -error * strength / total);
  }
}

function solveNodeGround() {
  for (const node of state.nodes) {
    if (node.mount || state.rotors.some((rotor) => rotor.nodeId === node.id && rotor.type === 'wheel')) continue;
    const radius = node.radius || 6;
    const floor = GROUND_Y - radius;
    if (node.y > floor) {
      node.y = floor;
      const vx = node.x - node.px;
      node.px = node.x - vx * .78;
      node.py = node.y + (node.y - node.py) * .16;
    }
  }
}

function shaftGroups() {
  const groups = new Map();
  for (const rotor of state.rotors) {
    if (!groups.has(rotor.nodeId)) groups.set(rotor.nodeId, []);
    groups.get(rotor.nodeId).push(rotor);
  }
  return groups;
}

function driveRotors(step) {
  const motors = state.attachments.filter((item) => item.type === 'motor');
  const groups = shaftGroups();
  for (const motor of motors) {
    const rotors = groups.get(motor.nodeId) || [];
    const target = state.drive * motor.amount * .055;
    for (const rotor of rotors) rotor.omega += (target - rotor.omega) * .22 * step;
  }
  for (const rotors of groups.values()) {
    if (rotors.length < 2) continue;
    const average = rotors.reduce((sum, rotor) => sum + rotor.omega, 0) / rotors.length;
    for (const rotor of rotors) rotor.omega += (average - rotor.omega) * .65;
  }
}

function solveGearCoupling(step) {
  const gears = state.rotors.filter((rotor) => rotor.type === 'gear');
  for (let i = 0; i < gears.length; i += 1) {
    const a = gears[i];
    const nodeA = nodeById(a.nodeId);
    if (!nodeA) continue;
    for (let j = i + 1; j < gears.length; j += 1) {
      const b = gears[j];
      if (a.nodeId === b.nodeId) continue;
      const nodeB = nodeById(b.nodeId);
      if (!nodeB) continue;
      const d = distance(nodeA, nodeB);
      const meshDistance = a.radius + b.radius;
      if (Math.abs(d - meshDistance) > 10) continue;
      const targetB = -a.omega * (a.radius / b.radius);
      const targetA = -b.omega * (b.radius / a.radius);
      b.omega += (targetB - b.omega) * .32 * step;
      a.omega += (targetA - a.omega) * .16 * step;
    }
  }
}

function solveWheelGround(step) {
  for (const wheel of state.rotors.filter((rotor) => rotor.type === 'wheel')) {
    const node = nodeById(wheel.nodeId);
    if (!node || node.fixed) continue;
    const floor = GROUND_Y - wheel.radius;
    if (node.y > floor) {
      node.y = floor;
      const currentVx = node.x - node.px;
      const surfaceVelocity = wheel.omega * wheel.radius * .75;
      const correction = clamp(surfaceVelocity - currentVx, -2.4, 2.4) * .3 * step;
      node.x += correction;
      node.px -= correction * .25;
      node.py = node.y + (node.y - node.py) * .12;
      if (Math.abs(state.drive) < .01) wheel.omega += ((currentVx / Math.max(10, wheel.radius)) - wheel.omega) * .3;
    }
  }
}

function updateRotorAngles(step) {
  for (const rotor of state.rotors) {
    rotor.omega *= rotor.type === 'wheel' ? .994 : .982;
    rotor.angle += rotor.omega * step;
  }
}

function updateWinches(step) {
  if (!activeWinchDirection) return;
  const winchNodes = new Set(state.rotors.filter((rotor) => rotor.type === 'winch').map((rotor) => rotor.nodeId));
  for (const rope of state.constraints.filter((item) => item.type === 'rope')) {
    if (!winchNodes.has(rope.a) && !winchNodes.has(rope.b)) continue;
    rope.maxLength = clamp(rope.maxLength - activeWinchDirection * 1.5 * step, 20, 700);
  }
}

function render() {
  resizeCanvas();
  if (state.mode === 'build') syncAllMounts(true);
  ctx.save();
  ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  drawBackground();
  drawGround();
  drawLayeredScene();
  drawSelection();
  drawPending();
  ctx.restore();
}

function drawLayeredScene() {
  const layers = [...new Set([
    ...state.constraints.map(itemLayer),
    ...state.rotors.map(itemLayer),
    ...state.nodes.map(itemLayer),
    ...state.joints.map(itemLayer)
  ])].sort((a, b) => a - b);
  if (!layers.length) layers.push(0);
  const activeLayer = selectionLayer();
  for (const layer of layers) {
    ctx.save();
    if (state.tool === 'layers' && state.selection && layer !== activeLayer) ctx.globalAlpha = .28;
    drawConstraints(layer);
    drawRotors(layer);
    drawNodesAndAttachments(layer);
    drawJoints(layer);
    ctx.restore();
  }
  if (state.tool === 'layers') {
    ctx.save();
    ctx.globalAlpha = .9;
    for (const node of state.nodes.filter((item) => item.mount)) drawNode(worldToScreen(node.x, node.y), node);
    drawJoints(null, true);
    ctx.restore();
  }
}
