function buildDrawbridgeV16() {
  clearWorld();
  const tower = addFixedBeamV16(320, 565, 320, 265, 0);
  const back = createNode(235, 565, { fixed: true });
  addBeam(back, tower.a).width = 15; addBeam(back, tower.b).width = 12;
  const pivotTower = mountNodeOnBeam(tower.beam, { x: 320, y: 455 }, { layer: 2 });
  const pivotBridge = createNode(pivotTower.x, pivotTower.y, { snap: false, layer: 2 });
  const tip = createNode(680, 455, { layer: 1 });
  const bridge = addBeam(pivotBridge, tip); bridge.width = 15; bridge.layer = 1;
  const joint = createJoint('limit', { node: pivotTower, beamId: tower.beam.id }, { node: pivotBridge, beamId: bridge.id });
  joint.type = 'limit'; joint.limitAngle = Math.PI * .48; joint.targetAngle = normalizeAngle(beamAngleV15(bridge) - beamAngleV15(tower.beam));
  const winchNode = createNode(320, 300, { fixed: true, layer: 3 });
  const winch = createRotor('winch', winchNode.id, { radius: 34, layer: 4 }); winch.mountMode = 'free';
  createAttachment('motor', winchNode.id, { amount: 2.4, layer: 5 });
  const rope = createConstraint('rope', winchNode.id, tip.id, { maxLength: distance(winchNode, tip), layer: 3 }); rope.maxLength = distance(winchNode, tip);
  const damper = createDamperV16(tower.a, tip); damper.damping = .48; damper.maxLength = 520; damper.minLength = 180; damper.layer = 0;
  state.camera = { x: 465, y: 410, zoom: .88 };
}

function installBlueprintsV16() {
  const list = document.querySelector('.blueprint-list');
  if (!list || list.querySelector('[data-blueprint="lift"]')) return;
  const definitions = [
    ['lift', 'Грузовой подъёмник', 'Каретки, толкатель, демпфер и контур A.'],
    ['press', 'Механический пресс', 'Жёсткая рама, направляющая и управляемая траверса.'],
    ['drawbridge', 'Подъёмный мост', 'Ограниченный шарнир, лебёдка, трос и демпфер.']
  ];
  for (const [id, title, description] of definitions) {
    const button = document.createElement('button');
    button.className = 'blueprint blueprint-v16'; button.dataset.blueprint = id; button.type = 'button'; button.dataset.nativePress = '';
    button.innerHTML = `<span class="blueprint-machine-v16 blueprint-${id}-v16" aria-hidden="true"></span><span><b></b><small></small></span>`;
    button.querySelector('b').textContent = title; button.querySelector('small').textContent = description;
    button.addEventListener('click', () => loadBlueprint(id));
    list.insertBefore(button, list.querySelector('[data-blueprint="blank"]'));
  }
}

const baseLoadBlueprintExpansionV16 = loadBlueprint;
loadBlueprint = function loadExpandedBlueprintV16(name) {
  if (!['lift', 'press', 'drawbridge'].includes(name)) return baseLoadBlueprintExpansionV16(name);
  pushUndo();
  if (name === 'lift') buildLiftV16();
  else if (name === 'press') buildPressV16();
  else buildDrawbridgeV16();
  setCurrentModelIdentity(null, name === 'lift' ? 'Грузовой подъёмник' : name === 'press' ? 'Механический пресс' : 'Подъёмный мост');
  closeBlueprints(); saveSoon(); updateInspector(); updateHistoryButtons(); frameConstruction();
  setYardNote(name === 'drawbridge' ? 'Мост готов. Удерживай кнопки лебёдки во время испытания.' : 'Механизм готов. Контур A приводит его в движение.', 2700);
  heavySound(145, .1); vibrate(14);
};
