const baseBuildUpgradeChoices = buildUpgradeChoices;
buildUpgradeChoices = function expandedUpgradeChoices() {
  const count = campaign?.currentRoute?.reward === 'upgrade' ? 4 : 3;
  const choices = getUpgradeChoices(campaign, count);
  dom.upgradeList.replaceChildren();

    const item = UPGRADES[key];
    const level = campaign.upgrades?.[key] || 0;
    const button = document.createElement('button');
    button.type = 'button';
    button.dataset.upgrade = key;
    button.setAttribute('data-native-press', '');
    button.innerHTML = `<i>${UPGRADE_ICONS[key] || '★'}</i><span><b>${item.name}${item.max > 1 ? ` · ${level + 1}/${item.max}` : ''}</b><small>${item.description}</small></span>`;
    dom.upgradeList.append(button);
  }
};

chooseDoctrine = function expandedChooseDoctrine(doctrine) {
  campaign = createCampaign(Date.now(), doctrine);
  migrateCampaignContent(campaign, profile.doctrineMastery[doctrine] || 0);
  battle = generateBattle(campaign);
  pending = 'route';

  profile.campaigns += 1;

chooseUpgrade = function expandedChooseUpgrade(key) {
  if (mode !== 'victory') return;
  inputLocked = true;
  completeVictory(campaign, battle, key || null);
  applyRouteReward();
  profile.victories += 1;
  profile.doctrineMastery[campaign.doctrine] = (profile.doctrineMastery[campaign.doctrine] || 0) + 1;
  profile.bestScore = Math.max(profile.bestScore || 0, campaign.score);
  saveJSON(PROFILE_KEY, profile);
  pending = shouldDraftOfficer() ? 'officer' : 'route';
  campaign.currentRoute = null;
  campaign.routeChoices = [];
  selectedId = null;
  inspectedId = null;
  inputLocked = false;
  persist();
  sound('stamp');
  if (pending === 'officer') showOfficerDraft();
  else showRouteMap();
};

function continueExpandedCampaign() {
  const saved = loadJSON(SAVE_KEY, null);
  if (!validSave(saved)) {
    toast('Сохранение не найдено.');
    refreshMenu();
    return;
  }
  campaign = migrateCampaignContent(saved.campaign, profile.doctrineMastery[saved.campaign.doctrine] || 0);
  battle = saved.battle;
  pending = saved.pending || battle.result || null;
  if (!battle.mission && !['route', 'officer'].includes(pending)) {
    const legacyRoute = {
      id: `legacy-${campaign.stage}`,
      title: 'Старый рубеж',
      mission: 'control',
      threat: 1,
      reward: 'renown',
      rewardValue: 12,
      faction: 'crown',
      commander: 'varr',
      note: 'Продолжение кампании версии 1.0.',
    };
    campaign.currentRoute = legacyRoute;
    applyRouteToBattle(battle, campaign, legacyRoute);
  }
  if (!pending && battle.phase === 'enemy') {
    const recoveredResult = finishRound(battle, campaign);
    if (recoveredResult) {
      battle.result = recoveredResult;
      pending = recoveredResult;
    }
  }
  selectedId = null;
  inspectedId = null;
  renderAll();
  if (pending === 'route') showRouteMap();
  else if (pending === 'event') showCampaignEvent(campaign.pendingEventId);
  else if (pending === 'officer') showOfficerDraft();
  else if (pending === 'victory') showVictory();
  else if (pending === 'defeat') {
    if (!campaign.history.some((entry) => entry.result === 'defeat' && entry.stage === campaign.stage)) completeDefeat(campaign, battle);
    showDefeat();
  } else showBriefing();
  sound('paper');
}

function showRouteMap() {
  migrateCampaignContent(campaign, profile.doctrineMastery[campaign.doctrine] || 0);
  mode = 'route';
  pending = 'route';
  if (!campaign.routeChoices.length) generateFrontChoices(campaign);
  exp.routeStage.textContent = campaign.stage;
  renderRouteChoices();
  showScreen('routeScreen');
  persist();
}

function selectFrontRoute(routeId) {
  const route = campaign.routeChoices.find((choice) => choice.id === routeId);
  if (!route) return;
  campaign.currentRoute = route;
  campaign.encounteredFactions = [...new Set([...campaign.encounteredFactions, route.faction])];
  campaign.encounteredCommanders = [...new Set([...campaign.encounteredCommanders, route.commander])];
  profile.factionsSeen = [...new Set([...profile.factionsSeen, route.faction])];
  profile.commandersSeen = [...new Set([...profile.commandersSeen, route.commander])];
  saveJSON(PROFILE_KEY, profile);
  battle = applyRouteToBattle(generateBattle(campaign), campaign, route);
  campaign.nextBattleModifiers = {};
  selectedId = null;
  inspectedId = null;
  const eventId = chooseCampaignEventId();
  if (eventId) {
    campaign.pendingEventId = eventId;
    pending = 'event';
    persist();
    showCampaignEvent(eventId);
  } else {
    campaign.pendingEventId = null;
    pending = null;
    persist();
    showBriefing();
  }
  sound('paper');
  haptic(12);
}

function chooseCampaignEventId() {
  const rng = seededRandom(hashSeed(campaign.seed, campaign.stage, campaign.currentRoute?.id, 'event-v2'));
  if (campaign.stage <= 1 || (campaign.stage % 2 !== 0 && rng() > .42)) return null;
  const ids = Object.keys(EXP_EVENTS);
  return ids[Math.floor(rng() * ids.length)];
}

function showCampaignEvent(eventId) {
  const event = EXP_EVENTS[eventId];
  if (!event) {
    pending = null;
    campaign.pendingEventId = null;
    persist();
    showBriefing();
    return;
  }
  mode = 'event';
  pending = 'event';
  campaign.pendingEventId = eventId;
  exp.eventMark.textContent = event.mark;
  exp.eventTitle.textContent = event.title;
  exp.eventCopy.textContent = event.copy;
  exp.eventConsequence.textContent = event.consequence;
  exp.eventChoices.replaceChildren();
  event.choices.forEach((choice, index) => {
    const button = document.createElement('button');
    button.type = 'button';
    button.dataset.eventChoice = String(index);
    button.setAttribute('data-native-press', '');
    button.innerHTML = `<span>${choice.title}</span><small>${choice.note}</small>`;
    exp.eventChoices.append(button);
  });
  showScreen('eventScreen');
  persist();
}

function applyEventChoice(index) {
  const event = EXP_EVENTS[campaign.pendingEventId];
  const choice = event?.choices?.[index];
  if (!choice) return;
  applyEventEffects(choice.effects || {});
  campaign.pendingEventId = null;
  pending = null;
  persist();
  sound('stamp');
  haptic(16);
  showBriefing();
}

function applyEventEffects(effects) {
  const players = battle.units.filter((unit) => unit.side === 'player' && !unit.routed && unit.hp > 0);
  campaign.renown = Math.max(0, campaign.renown + (effects.renown || 0));
  campaign.resources.intel = Math.max(0, campaign.resources.intel + (effects.intel || 0));
  campaign.resources.requisition = Math.max(0, campaign.resources.requisition + (effects.requisition || 0));
  if (effects.command) { battle.command += effects.command; battle.maxCommand += effects.command; }
  if (effects.morale) players.forEach((unit) => { unit.morale = clamp(unit.morale + effects.morale, 0, unit.maxMorale); });
  if (effects.supply) players.forEach((unit) => { unit.supply = clamp(unit.supply + effects.supply, 0, unit.maxSupply); });
  if (effects.health) players.forEach((unit) => { unit.maxHp += effects.health; unit.hp += effects.health; });
  if (effects.turnLimit) battle.turnLimit = clamp(battle.turnLimit + effects.turnLimit, 7, 16);
  if (effects.playerResolve) battle.playerResolve = clamp(battle.playerResolve + effects.playerResolve, 1, 120);
  if (effects.enemyResolve) battle.enemyResolve = clamp(battle.enemyResolve + effects.enemyResolve, 10, 150);
  if (effects.weather && WEATHER[effects.weather]) battle.weather = effects.weather;
}

function shouldDraftOfficer() {
  const routeOfficer = battle?.route?.reward === 'officer';
  return routeOfficer || campaign.victories % 3 === 0;
}

function showOfficerDraft() {
  mode = 'officer';
  pending = 'officer';
  const choices = generateOfficerChoices(campaign, 3);
  if (!choices.length) {
    campaign.renown += 25;
    pending = 'route';
    persist();
    showRouteMap();
    return;
  }
  exp.officerList.replaceChildren();
  for (const id of choices) {
    const officer = OFFICERS[id];
    const button = document.createElement('button');
    button.type = 'button';
    button.dataset.officer = id;
    button.setAttribute('data-native-press', '');
    button.innerHTML = `<i>${officer.mark}</i><span><b>${officer.name}</b><em>${officer.role}</em><small>${officer.effect}</small></span>`;
    exp.officerList.append(button);
  }
  showScreen('officerScreen');
  persist();
}

function chooseOfficer(officerId) {
  if (!addOfficer(campaign, officerId)) return;
  pending = 'route';
  persist();
  sound('victory');
  haptic([18, 28, 35]);
  showRouteMap();
}

function applyRouteReward() {
  const route = battle?.route;
  if (!route) return;
  if (battle.routeRewardClaimed) return;
  battle.routeRewardClaimed = true;
  if (route.reward === 'renown') campaign.renown += route.rewardValue || 15;
  if (route.reward === 'requisition') campaign.resources.requisition += route.rewardValue || 1;
  if (route.reward === 'intel') campaign.resources.intel += 1;
  if (route.reward === 'upgrade') {
    campaign.renown += 10;
    campaign.nextBattleModifiers.command = (campaign.nextBattleModifiers.command || 0) + 1;
  }
}
