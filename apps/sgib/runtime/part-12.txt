  dom.hapticsPauseBtn.addEventListener('click', toggleHaptics);

  dom.pauseBtn.addEventListener('click', () => { pressFeedback(); pauseGame(); });
  dom.resumeBtn.addEventListener('click', () => { pressFeedback(); resumeGame(); });
  dom.restartBtn.addEventListener('click', () => { pressFeedback(); restartCurrent(); });
  dom.pauseMenuBtn.addEventListener('click', () => { pressFeedback(); returnToMenu(); });
  dom.undoBtn.addEventListener('click', () => { pressFeedback(); undoFold(); });

  dom.nextBtn.addEventListener('click', () => {
    pressFeedback();
    if (state.mode === 'endless') {
      state.endlessSeed = (state.endlessSeed + 104729) >>> 0;
      settings.endlessSeed = state.endlessSeed;
      store.set('endlessSeed', state.endlessSeed);
      loadEndlessSheet();
    } else if (state.levelIndex < LEVELS.length - 1) {
      startCampaign(state.levelIndex + 1);
    } else {
      startEndless(true);
    }
  });
  dom.resultRetryBtn.addEventListener('click', () => {
    pressFeedback();
    if (state.mode === 'endless') startEndless(true);
    else restartCurrent();
  });
  dom.resultMenuBtn.addEventListener('click', () => { pressFeedback(); returnToMenu(); });

  dom.failRetryBtn.addEventListener('click', () => {
    pressFeedback();
    if (state.mode === 'endless') startEndless(true);
    else restartCurrent();
  });
  dom.failUndoBtn.addEventListener('click', () => { pressFeedback(); undoFold(); });
  dom.failMenuBtn.addEventListener('click', () => { pressFeedback(); returnToMenu(); });
  dom.reloadBtn.addEventListener('click', () => location.reload());

  const handleViewportMutation = () => {
    cancelGesture();
    resizeCanvas();
  };

  document.addEventListener('visibilitychange', () => {
    if (document.hidden) {
      cancelGesture();
      audio.context?.suspend().catch(() => {});
    }
  });
  window.addEventListener('appviewportchange', handleViewportMutation);
  window.addEventListener('resize', handleViewportMutation);
  window.addEventListener('orientationchange', () => {
    cancelGesture();
    setTimeout(resizeCanvas, 50);
  });
  window.addEventListener('blur', () => cancelGesture());
  window.addEventListener('pagehide', () => cancelGesture());
  window.addEventListener('keydown', (event) => {
    if (event.key === 'Escape' && state.mode !== 'menu') {
      if (state.screen === 'pauseScreen') resumeGame();
      else if (!state.screen) pauseGame();
    }
    if ((event.ctrlKey || event.metaKey) && event.key.toLowerCase() === 'z') {
      event.preventDefault();
      undoFold();
    }
  });
}

function collectDom() {
  const ids = [
    'app', 'paperCanvas', 'gameHud', 'modeLabel', 'levelTitle', 'foldCounter', 'pauseBtn', 'undoBtn',
    'gestureHint', 'gestureHintText', 'menuScreen', 'continueBtn', 'continueLabel', 'continueMeta',
    'levelsBtn', 'progressMeta', 'endlessBtn', 'endlessMeta', 'soundMenuBtn', 'levelsScreen',
    'closeLevelsBtn', 'levelGrid', 'pauseScreen', 'resumeBtn', 'restartBtn', 'soundPauseBtn',
    'hapticsPauseBtn', 'pauseMenuBtn', 'resultScreen', 'resultStamp', 'resultTitle', 'resultCopy',
    'resultFolds', 'resultUnit', 'nextBtn', 'nextLabel', 'nextMeta', 'resultRetryBtn', 'resultMenuBtn',
    'failScreen', 'failCopy', 'failRetryBtn', 'failUndoBtn', 'failMenuBtn', 'errorScreen', 'errorCopy',
    'reloadBtn', 'toast'
  ];
  for (const id of ids) {
    const element = document.getElementById(id);
    if (!element) throw new Error(`Missing required element #${id}`);
    dom[id] = element;
  }
  canvas = dom.paperCanvas;
  context = canvas.getContext('2d', { alpha: false, desynchronized: true });
  if (!context) throw new Error('Canvas 2D is unavailable');
}

function validateStoredState(value) {
  return value && typeof value === 'object'
    && Number.isInteger(value.unlocked)
    && value.unlocked >= 1
    && value.best && typeof value.best === 'object'
    && typeof value.sound === 'boolean'
    && typeof value.haptics === 'boolean'
    && Number.isFinite(value.endlessBest)
    && Number.isInteger(value.lastLevel)
    && Number.isFinite(value.endlessSeed);
}
