  dom.toast.textContent = message;
  dom.toast.classList.add('is-visible');
  toastTimer = setTimeout(() => dom.toast.classList.remove('is-visible'), 1500);
}

function snapshotPointerFrame() {
  return {
    scale: Number.isFinite(view.scale) && view.scale > 0.0001 ? view.scale : 1,
    offsetX: Number.isFinite(view.offsetX) ? view.offsetX : 0,
    offsetY: Number.isFinite(view.offsetY) ? view.offsetY : 0
  };
}

function pointerClientPoint(event) {
  const x = Number(event.clientX);
  const y = Number(event.clientY);
  return Number.isFinite(x) && Number.isFinite(y) ? { x, y } : null;
}

function pointerToLogical(event, frame = view) {
  const client = pointerClientPoint(event);
  if (!client) return null;
  const scale = Number.isFinite(frame?.scale) && frame.scale > 0.0001 ? frame.scale : 1;
  const offsetX = Number.isFinite(frame?.offsetX) ? frame.offsetX : 0;
  const offsetY = Number.isFinite(frame?.offsetY) ? frame.offsetY : 0;
  return {
    x: (client.x - offsetX) / scale,
    y: (client.y - offsetY) / scale
  };
}

function foldCommitDistance(pointerType) {
  if (pointerType === 'touch') return 54;
  if (pointerType === 'pen') return 38;
  return 24;
}

function pointOnPaper(point) {
  return Boolean(point) && state.layers.some((layer) => pointInPolygon(point, layer.polygon));
}

function tryCapturePointer(pointerId) {
  try {
    canvas.setPointerCapture?.(pointerId);
  } catch (error) {
    console.warn('СГИБ pointer capture was unavailable', error);
  }
}

function safeReleasePointer(pointerId) {
  try {
    if (typeof canvas.hasPointerCapture !== 'function' || canvas.hasPointerCapture(pointerId)) {
      canvas.releasePointerCapture?.(pointerId);
    }
  } catch (error) {
    console.warn('СГИБ pointer release was unavailable', error);
  }
}

function beginGesture(event) {
  if (state.mode === 'menu' || state.screen || state.animation || state.resolved || state.paused) return;
  if (state.gesture || event.isPrimary === false) return;
  if (event.pointerType === 'mouse' && event.button !== 0) return;

  const client = pointerClientPoint(event);
  if (!client) return;
  const coordinateFrame = snapshotPointerFrame();
  const point = pointerToLogical(event, coordinateFrame);
  if (!pointOnPaper(point)) {
    showToast('Начни жест на бумаге');
    return;
  }

  event.preventDefault?.();
  audio.unlock();
  state.gesture = {
    pointerId: event.pointerId,
    pointerType: event.pointerType || 'touch',
    coordinateFrame,
    startClient: client,
    currentClient: client,
    dragDistance: 0,
    commitDistance: foldCommitDistance(event.pointerType),
    start: point,
    current: point,
    line: null,
    angle: 0,
    valid: false
  };
  state.lastInputAt = performance.now();
  hideHint();
  tryCapturePointer(event.pointerId);
}

function moveGesture(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;
  const client = pointerClientPoint(event);
  const point = pointerToLogical(event, gesture.coordinateFrame);
  if (!client || !point) return;

  event.preventDefault?.();
  gesture.currentClient = client;
  gesture.dragDistance = Math.hypot(
    client.x - gesture.startClient.x,
    client.y - gesture.startClient.y
  );
  gesture.current = point;
  gesture.line = lineFromGesture(gesture.start, point);

  if (!gesture.line || gesture.dragDistance < 8) {
    gesture.angle = 0;
    gesture.valid = false;
    return;
  }

  // Progress is based on real CSS pixels. A visualViewport resize can no longer
  // turn a stationary finger into a full logical drag.
  const rawProgress = clamp((gesture.dragDistance - 8) / 240, 0, 1);
  const easedProgress = rawProgress * rawProgress * (3 - 2 * rawProgress);
  gesture.angle = Math.min(easedProgress, 0.86) * Math.PI;
  gesture.valid = gesture.dragDistance >= gesture.commitDistance && validateFold(gesture.line);
}

function endGesture(event) {
  const gesture = state.gesture;
  if (!gesture || gesture.pointerId !== event.pointerId) return;

  // Safari may dispatch lostpointercapture synchronously from releasePointerCapture().
  // Detach the gesture first so that event can never erase the snapshot used below.
  state.gesture = null;
  safeReleasePointer(event.pointerId);
  state.lastInputAt = performance.now();

  const dragDistance = Number.isFinite(gesture.dragDistance) ? gesture.dragDistance : 0;
  if (!gesture.line || dragDistance < gesture.commitDistance || !gesture.valid) {
    if (dragDistance < 8) {
      showHint(state.level?.hint || 'Потяни по листу и отпусти', true);
      return;
    }
    audio.invalid();
    vibrate(14);
    showToast(
      dragDistance < gesture.commitDistance
        ? 'Потяни дальше, затем отпусти'
        : 'Сгиб должен пересечь лист'
    );
    return;
  }
  commitFold(gesture.line, gesture.angle);
}

function cancelGesture(event) {
  const gesture = state.gesture;
  if (!gesture || (event && gesture.pointerId !== event.pointerId)) return;
  state.gesture = null;
  safeReleasePointer(gesture.pointerId);
}

function validateFold(line) {
  if (!line || line.length < 44) return false;
  let totalArea = 0;
  let movingArea = 0;
  let stationaryArea = 0;
  let crosses = false;
  for (const layer of state.layers) {
    const area = polygonArea(layer.polygon);
    const moving = clipPolygon(layer.polygon, line, false);
    const stationary = clipPolygon(layer.polygon, line, true);
    totalArea += area;
    movingArea += polygonArea(moving);
    stationaryArea += polygonArea(stationary);
    if (lineCrossesPolygon(line, layer.polygon)) crosses = true;
  }
  const minimumArea = Math.max(850, totalArea * 0.006);
  return crosses && movingArea > minimumArea && stationaryArea > minimumArea;
}

function commitFold(line, currentAngle) {
  state.history.push(cloneSnapshot());
  const safeAngle = Number.isFinite(currentAngle) ? currentAngle : 0;
  const settledAngle = clamp(safeAngle, 0.06 * Math.PI, 0.86 * Math.PI);
  state.animation = {
    line,
    fromAngle: settledAngle,
    angle: settledAngle,
    startedAt: performance.now(),
    duration: REDUCED_MOTION ? 60 : 380,
    applied: false
  };
  state.lastInputAt = performance.now();
  audio.fold();
  vibrate(8);
  updateHud();
}

function applyFold(line) {
  const nextLayers = [];
  let highestDepth = state.layers.reduce((maximum, layer) => Math.max(maximum, layer.depth || 0), 0);
  for (const layer of state.layers) {
    const stationary = clipPolygon(layer.polygon, line, true);
    const moving = clipPolygon(layer.polygon, line, false);
    if (stationary.length >= 3 && polygonArea(stationary) > 1) {
      nextLayers.push({ ...layer, polygon: stationary });
    }
    if (moving.length >= 3 && polygonArea(moving) > 1) {
      highestDepth += 1;
      nextLayers.push({ polygon: reflectPolygon(moving, line), depth: highestDepth, shade: (layer.shade || 0) + 1 });
    }
