function drawMenuScene() {
  context.save();
  const centerX = LOGICAL_WIDTH * 0.43;
  const centerY = LOGICAL_HEIGHT * 0.57;
  context.translate(centerX, centerY);
  context.rotate(-0.09);
  const sheets = [
    { offsetX: -65, offsetY: 52, angle: -0.04, shade: 4 },
    { offsetX: 20, offsetY: 24, angle: 0.055, shade: 2 },
    { offsetX: -10, offsetY: -12, angle: -0.015, shade: 0 }
  ];
  for (const sheet of sheets) {
    context.save();
    context.translate(sheet.offsetX, sheet.offsetY);
    context.rotate(sheet.angle);
    const polygon = [
      { x: -315, y: -440 },
      { x: 310, y: -430 },
      { x: 325, y: 430 },
      { x: -320, y: 445 }
    ];
    drawPaperPolygon(polygon, { depth: sheet.shade, shade: sheet.shade });
    context.restore();
  }
  context.restore();

  context.save();
  context.globalAlpha = 0.55;
  const demoInk = { id: 'menu-ink', type: 'ink', radius: 27, x: 260, y: 1000 };
  const demoVoid = { id: 'menu-void', type: 'void', radius: 32, x: 730, y: 710 };
  drawEntity(demoInk);
  drawEntity(demoVoid);
  context.setLineDash([10, 12]);
  context.strokeStyle = 'rgba(49,93,80,.45)';
  context.lineWidth = 2;
  context.beginPath();
  context.moveTo(260, 1000);
  context.lineTo(730, 710);
  context.stroke();
  context.restore();
}

function recoverFailedFold(error) {
  console.error('СГИБ fold application failed', error);
  clearTimeout(completionTimer);
  const snapshot = state.history.pop();
  if (snapshot) restoreSnapshot(snapshot);
  else restartCurrent();
  state.animation = null;
  state.resolved = false;
  state.paused = false;
  showScreen(null);
  showToast('Сгиб отменён — лист восстановлен');
  audio.invalid();
  vibrate([18, 35, 10]);
}

function updateAnimation(now) {
  const animation = state.animation;
  if (!animation) return;

  const duration = Number.isFinite(animation.duration) && animation.duration > 0 ? animation.duration : 1;
  const startedAt = Number.isFinite(animation.startedAt) ? animation.startedAt : now - duration;
  const progress = clamp((now - startedAt) / duration, 0, 1);
  const eased = 1 - Math.pow(1 - progress, 3);
  const fromAngle = Number.isFinite(animation.fromAngle) ? animation.fromAngle : 0;
  animation.angle = fromAngle + (Math.PI - fromAngle) * eased;

  if (progress >= 1 && !animation.applied) {
    animation.applied = true;
    state.animation = null;
    try {
      applyFold(animation.line);
    } catch (error) {
      recoverFailedFold(error);
    }
  }
}

function updateEffects(deltaSeconds) {
  for (const particle of state.particles) {
    particle.age += deltaSeconds;
    particle.x += particle.vx * deltaSeconds;
    particle.y += particle.vy * deltaSeconds;
    particle.vx *= Math.pow(0.08, deltaSeconds);
    particle.vy = particle.vy * Math.pow(0.12, deltaSeconds) + 70 * deltaSeconds;
  }
  state.particles = state.particles.filter((particle) => particle.age < particle.life);
  for (const pulse of state.pulses) pulse.age += deltaSeconds;
  state.pulses = state.pulses.filter((pulse) => pulse.age < pulse.life);
}

function recoverStaleInteractions(now) {
  const gesture = state.gesture;
  const lastGestureEvent = gesture?.lastEventAt || gesture?.startedAt;
  if (gesture && Number.isFinite(lastGestureEvent) && now - lastGestureEvent > GESTURE_STALE_MS) {
    cancelGesture();
    showHint(state.level?.hint || 'Потяни по листу и отпусти', true);
  }

  const animation = state.animation;
  const animationStartedAt = animation?.startedAt;
  if (animation && Number.isFinite(animationStartedAt) && now - animationStartedAt > ANIMATION_STALE_MS) {
    recoverFailedFold(new Error('Fold animation watchdog expired'));
  }
}

function render(now) {
  const frameNow = Number.isFinite(now) ? now : performance.now();
  try {
    const deltaSeconds = clamp((frameNow - lastFrameTime) / 1000, 0, 0.05);
    lastFrameTime = frameNow;
    updateAnimation(frameNow);
    recoverStaleInteractions(frameNow);
    updateEffects(deltaSeconds);
    resizeCanvas();
    beginLogicalFrame();
    if (state.mode === 'menu') drawMenuScene();
    else drawPaperScene();
    context.setTransform(1, 0, 0, 1, 0, 0);
  } catch (error) {
    console.error('СГИБ frame failed', error);
    if (state.animation) {
      try {
        recoverFailedFold(error);
      } catch (recoveryError) {
        console.error('СГИБ frame recovery failed', recoveryError);
        state.animation = null;
      }
    }
  } finally {
    // A single malformed preview frame must never kill the RAF loop and leave
    // the game permanently blocked by state.animation.
    rafId = requestAnimationFrame(render);
  }
}

function bindControls() {
  canvas.addEventListener('pointerdown', beginGesture, { passive: false });

  // Touch and pen input use the window path instead of explicit pointer capture.
  // This avoids WebKit capture-loss races while keeping the gesture alive if the
  // finger crosses UI chrome or the canvas boundary.
  window.addEventListener('pointermove', moveGesture, { capture: true, passive: false });
  window.addEventListener('pointerup', endGesture, { capture: true, passive: false });
  window.addEventListener('pointercancel', cancelGesture, { capture: true, passive: false });

  canvas.addEventListener('lostpointercapture', (event) => {
    const gesture = state.gesture;
    if (!gesture || !gesture.captured || gesture.pointerId !== event.pointerId) return;
    if (cancelGesture(event)) {
      showHint(state.level?.hint || 'Потяни по листу и отпусти', true);
    }
  });

  dom.continueBtn.addEventListener('click', () => {
    pressFeedback();
    const unlocked = clamp(settings.unlocked, 1, LEVELS.length);
    startCampaign(clamp(settings.lastLevel, 0, unlocked - 1));
  });
  dom.levelsBtn.addEventListener('click', () => { pressFeedback(); buildLevelGrid(); showScreen('levelsScreen'); });
  dom.closeLevelsBtn.addEventListener('click', () => { pressFeedback(); showScreen('menuScreen'); });
  dom.endlessBtn.addEventListener('click', () => { pressFeedback(); startEndless(true); });
  dom.soundMenuBtn.addEventListener('click', toggleSound);
  dom.soundPauseBtn.addEventListener('click', toggleSound);
