    const end = { x: midpoint.x + normal.x * 285, y: midpoint.y + normal.y * 285 };
    const line = lineFromGesture(start, end);
    const entities = [];
    const occupied = [];
    let created = 0;

    for (let sample = 0; sample < 300 && created < pairCount; sample += 1) {
      const point = normalizedPoint(0.13 + random() * 0.74, 0.12 + random() * 0.76);
      if (signedDistance(point, line) > -78) continue;
      const partner = reflectPoint(point, line);
      if (!pointInPolygon(partner, BASE_PAPER)) continue;
      if (occupied.some((other) => distance(other, point) < 88 || distance(other, partner) < 88)) continue;
      const pairType = ['ink', 'void'];
      entities.push({ id: `endless-${created}-a`, type: pairType[0], x: point.x, y: point.y, radius: 25 });
      entities.push({ id: `endless-${created}-b`, type: pairType[1], x: partner.x, y: partner.y, radius: 30 });
      occupied.push(point, partner);
      created += 1;
    }

    if (created === pairCount) {
      return {
        title: `Стопка ${streak + 1}`,
        note: `${pairCount} ${pairCount === 1 ? 'совпадение' : pairCount < 5 ? 'совпадения' : 'совпадений'}`,
        maxFolds: 1,
        hint: 'На каждом листе есть один точный сгиб',
        hintFromPoint: start,
        hintToPoint: end,
        entities
      };
    }
  }
  throw new Error('Не удалось создать решаемый лист');
}

function mulberry32(seed) {
  let value = seed >>> 0;
  return () => {
    value += 0x6D2B79F5;
    let t = value;
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}

function showScreen(id) {
  state.screen = id;
  for (const screen of document.querySelectorAll('.screen')) {
    screen.classList.toggle('is-visible', screen.id === id);
  }
}

function setPlaying(value) {
  dom.app.classList.toggle('is-playing', value);
  if (!value) dom.gestureHint.classList.remove('is-visible');
}

function startCampaign(index) {
  clearTimeout(completionTimer);
  const safeIndex = clamp(index, 0, LEVELS.length - 1);
  const level = LEVELS[safeIndex];
  state.mode = 'campaign';
  state.levelIndex = safeIndex;
  state.level = level;
  state.layers = createInitialLayers();
  state.entities = buildEntities(level);
  state.creases = [];
  state.foldsUsed = 0;
  state.maxFolds = level.maxFolds;
  state.history = [];
  state.gesture = null;
  state.animation = null;
  state.particles = [];
  state.pulses = [];
  state.resolved = false;
  state.paused = false;
  state.lastInputAt = performance.now();
  settings.lastLevel = safeIndex;
  store.set('lastLevel', safeIndex);
  setPlaying(true);
  showScreen(null);
  updateHud();
  showHint(level.hint, true);
}

function startEndless(resetStreak = true) {
  clearTimeout(completionTimer);
  state.mode = 'endless';
  if (resetStreak) state.endlessStreak = 0;
  state.endlessSeed = (settings.endlessSeed + 1 + state.endlessStreak * 7919) >>> 0;
  settings.endlessSeed = state.endlessSeed;
  store.set('endlessSeed', state.endlessSeed);
  loadEndlessSheet();
}

function loadEndlessSheet() {
  const level = buildEndlessLevel(state.endlessSeed, state.endlessStreak);
  state.level = level;
  state.layers = createInitialLayers();
  state.entities = level.entities.map((entity) => ({ ...entity }));
  state.creases = [];
  state.foldsUsed = 0;
  state.maxFolds = 1;
  state.history = [];
  state.gesture = null;
  state.animation = null;
  state.particles = [];
  state.pulses = [];
  state.resolved = false;
  state.paused = false;
  state.lastInputAt = performance.now();
  setPlaying(true);
  showScreen(null);
  updateHud();
  showHint(level.hint, true);
}
