function renderProvincePath(province) {
  const selected = selectedProvince === province.id;
  const threatened = province.owner !== 'player' && state.wars[relationKey(province.owner)] && adjacentToOwner(province, 'player');

}

function renderOwnedProvinceActions(province) {
  const nextBuilding = Object.entries(BUILDINGS).filter(([id]) => !province.buildings.includes(id));

function renderEnemyProvinceActions(province) {
  const key = relationKey(province.owner);
  const border = adjacentToOwner(province, 'player');
  return `<button class="drawer-action primary" data-province-action="attack" data-native-press ${!state.wars[key] || !border || totalArmy() < 25 ? 'disabled' : ''}><b>Начать поход</b><small>${!state.wars[key] ? 'Сначала объявите войну' : !border ? 'Нет общей границы' : 'Развернуть три фланга'}</small></button>`;
}

function disabled(cost) {
  return state.actions <= 0 || !canAfford(cost) ? 'disabled' : '';
}

function closeDrawer() {
  drawerOpen = false;
  selectedProvince = null;
  app.querySelector('#drawer')?.classList.remove('open');
  app.querySelectorAll('.province-shape').forEach((path) => path.classList.remove('selected'));
}

function provinceAction(province, action, buildingId) {
  if (action === 'attack') {
    state.battle = createBattle(province, false);
    drawerOpen = false;
    selectedProvince = null;
    tone('war');
    render();
    return;
  }
  if (!spendAction()) return;
  if (action === 'develop' && canAfford({ gold: 22 })) {
    spend({ gold: 22 });
    province.development += 1;
    province.population += 2;
    province.loyalty = clamp(province.loyalty + 2, 0, 100);
    addLog(`${province.name} получила новые мастерские и поселения.`);
    toast('Область развита');
  } else if (action === 'control') {
    const cost = state.techs.includes('roads') ? 8 : 12;
    if (!canAfford({ authority: cost })) return;
    spend({ authority: cost });
    province.autonomy = clamp(province.autonomy - 10, 0, 90);
    province.loyalty = clamp(province.loyalty - 3, 0, 100);
    state.estates.nobles.loyalty -= 2;
    addLog(`В области ${province.name} усилена княжеская администрация.`);
    toast('Автономия снижена');
  } else if (action === 'appease' && canAfford({ gold: 12, food: 7 })) {
    spend({ gold: 12, food: 7 });
    province.loyalty = clamp(province.loyalty + 12, 0, 100);
    province.unrest = Math.max(0, province.unrest - 2);
    addLog(`Праздник примирил жителей области ${province.name}.`);
    toast('Лояльность выросла');
  } else if (action === 'build' && BUILDINGS[buildingId] && !province.buildings.includes(buildingId)) {
    const building = BUILDINGS[buildingId];
    if (!canAfford({ gold: building.cost })) return;
    spend({ gold: building.cost });
    province.buildings.push(buildingId);
    if (buildingId === 'road') province.infrastructure += 1;
    addLog(`В области ${province.name} построено: ${building.name}.`);
    toast('Строительство завершено');
  }
  tone('success');
  drawerOpen = false;
  selectedProvince = null;
  render();
}
