function d25RenderPeace() {
  const wars = ['north', 'south'].filter((enemy) => state.wars[enemy]);
  if (!wars.length) return '<div class="quiet-note">Действующих внешних войн нет.</div>';
  return `<div class="d25-peace-list">${wars.map((enemy) => {
    const record = d25WarRecord(state, enemy);
    const remaining = state.provinces.filter((province) => province.owner === enemy).length;
    return `<article class="d25-peace-card"><header><div><h3>${d25Esc(enemyName(enemy))}</h3><p>${record.captured.length} занятых земель · ${remaining} вражеских областей</p></div><b class="${record.score < 0 ? 'negative' : ''}">${record.score >= 0 ? '+' : ''}${round(record.score)}<small>счёт войны</small></b></header>${d25Bar(record.exhaustion, 'Усталость от войны', record.exhaustion > 70 ? 'danger' : '')}<div class="d25-peace-actions"><button data-peace-action="white" data-enemy="${enemy}" data-native-press ${record.score < -8 ? 'disabled' : ''}><b>Белый мир</b><span>вернуть занятые земли</span></button><button data-peace-action="recognize" data-enemy="${enemy}" data-native-press ${record.score < 20 || !record.captured.length ? 'disabled' : ''}><b>Закрепить границы</b><span>сохранить завоевания · счёт 20</span></button><button data-peace-action="tribute" data-enemy="${enemy}" data-native-press ${record.score < 38 ? 'disabled' : ''}><b>Потребовать дань</b><span>завоевания + золото · счёт 38</span></button><button data-peace-action="subjugate" data-enemy="${enemy}" data-native-press ${record.score < 75 || remaining > 3 ? 'disabled' : ''}><b>Подчинить корону</b><span>вассальная держава · счёт 75</span></button></div></article>`;
  }).join('')}</div>`;
}

function d25RealmCard(enemy) {
  const relation = state.relations[enemy];
  const atWar = state.wars[enemy];
  const record = d25WarRecord(state, enemy);
  const provinces = state.provinces.filter((province) => province.owner === enemy).length;
  return `<article class="d25-realm ${atWar ? 'at-war' : ''}"><header><div><h3>${d25Esc(enemyName(enemy))}</h3><p>${provinces} областей · отношения ${round(relation)}</p></div><b>${atWar ? `ВОЙНА ${record.score >= 0 ? '+' : ''}${round(record.score)}` : relation > 35 ? 'БЛАГОСКЛОНЕН' : relation < -35 ? 'ВРАЖДЕБЕН' : 'НАСТОРОЖЕН'}</b></header><div><button data-diplomacy="${enemy}" data-native-press ${atWar || state.actions <= 0 || state.gold < 12 ? 'disabled' : ''}>Посольство <small>¤12</small></button>${atWar ? '<span>Мирные условия находятся выше</span>' : `<button class="danger" data-war="${enemy}" data-native-press ${state.actions <= 0 ? 'disabled' : ''}>Объявить войну</button>`}</div></article>`;
}

};

};
