    gain.gain.exponentialRampToValueAtTime(0.0001, now + 0.2);
    oscillator.connect(gain).connect(audioContext.destination);
    oscillator.start(now);
    oscillator.stop(now + 0.22);
  } catch {}
}

function toast(message) {
  let element = document.querySelector('.toast');
  if (!element) {
    element = document.createElement('div');
    element.className = 'toast';
    document.body.append(element);
  }
  element.textContent = message;
  requestAnimationFrame(() => element.classList.add('show'));
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => element.classList.remove('show'), 1900);
}

function spendAction() {
  if (state.actions <= 0) {
    toast('Решения этого сезона закончились');

}

function canAfford({ gold = 0, food = 0, manpower = 0, authority = 0, renown = 0 }) {
  return state.gold >= gold && state.food >= food && state.manpower >= manpower && state.authority >= authority && state.renown >= renown;
}

function spend(cost) {
  for (const key of ['gold', 'food', 'manpower', 'authority', 'renown']) state[key] -= cost[key] || 0;
}

function playerProvinces() {
  return state.provinces.filter((province) => province.owner === 'player');
}

function adjacentToOwner(province, owner) {
  return (ADJACENCY[province.id] || []).some((id) => provinceById(id)?.owner === owner);
}

function lawChoice(id) {
  const law = LAWS.find((item) => item.id === id);
  return law?.choices[state.laws[id] ?? 0];
}

function lawEffect(key) {
  return LAWS.reduce((sum, law) => sum + (lawChoice(law.id)?.effects?.[key] || 0), 0);
}

function buildingEffect(province, key) {
  return (province.buildings || []).reduce((sum, id) => sum + (BUILDINGS[id]?.[key] || 0), 0);
}

function provinceYield(province) {
  const terrain = TERRAIN[province.terrain];
  const resource = RESOURCES[province.resource][2];
  const control = clamp((100 - province.autonomy) / 100, 0.15, 1);
  const loyalty = clamp(province.loyalty / 100, 0.25, 1);
  const populationScale = province.population / 34;
  let tax = (terrain.tax + province.development * 0.8 + (resource.tax || 0) + buildingEffect(province, 'tax')) * control * loyalty * populationScale;
  let food = (terrain.food + (resource.food || 0) + buildingEffect(province, 'food')) * loyalty * populationScale;
  let manpower = (terrain.manpower + (resource.manpower || 0) + buildingEffect(province, 'manpower')) * control * populationScale;
  if (state.techs.includes('irrigation') && ['plains', 'river', 'delta'].includes(province.terrain)) food += 2;
  if (state.techs.includes('mills') && province.buildings.includes('fields')) { food += 2; tax += 1; }
  if (state.techs.includes('ledgers') && (province.buildings.includes('market') || resource.tax)) tax += 1.5;
  if (state.estates.burghers.privilege && province.buildings.includes('market')) tax += 1;
  if (state.estates.commons.privilege && province.buildings.includes('fields')) food += 1.5;
  return { tax, food, manpower };
}

function calculateEconomy() {
  const income = { tax: 0, food: 0, manpower: 0 };
  for (const province of playerProvinces()) {
    const yieldData = provinceYield(province);
    income.tax += yieldData.tax;
    income.food += yieldData.food;
    income.manpower += yieldData.manpower;
  }
  income.tax += lawEffect('tax') + state.ruler.stewardship;
  income.food += lawEffect('food');
  income.manpower += lawEffect('manpower');

  const armyUpkeep = totalArmy() * 0.085 + lawEffect('upkeep');
  const fortUpkeep = playerProvinces().reduce((sum, province) => sum + province.fort * 0.42, 0);
  const courtUpkeep = 4 + state.ruler.stewardship * 0.35 + lawEffect('court');
  const foodUse = totalArmy() * 0.055 + playerProvinces().reduce((sum, province) => sum + province.population * 0.035, 0);
  const net = {
    gold: income.tax - armyUpkeep - fortUpkeep - courtUpkeep,
    food: income.food - foodUse,
    manpower: income.manpower
  };
  return {
    income,
    expenses: { armyUpkeep, fortUpkeep, courtUpkeep, foodUse },
    net
  };
}

}
