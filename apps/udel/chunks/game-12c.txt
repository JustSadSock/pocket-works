  return `<button class="drawer-action primary" data-campaign-target="${province.id}" data-native-press ${!state.wars[enemy] || !route || state.campaign || totalArmy() < 20 || state.actions <= 0 || state.gold < 10 || state.food < 8 ? 'disabled' : ''}><b>Подготовить поход</b><small>${!state.wars[enemy] ? 'Сначала объявите войну' : !route ? 'Нет доступного маршрута' : state.campaign ? 'Другая дружина уже в походе' : `${route.length - 1} перехода · ¤10 ✦8`}</small></button>`;
};

const d25RenderProvinceDrawer2 = renderProvinceDrawer;
renderProvinceDrawer = function renderProvinceDrawerD25(province) {
  d25RenderProvinceDrawer2(province);
  const drawer = app.querySelector('#drawer');
  if (!drawer || !province) return;
  const campaign = state.campaign;
  if (campaign?.route?.includes(province.id)) {
    const isCurrent = d25CampaignPosition(campaign)?.id === province.id;
    drawer.querySelector('.drawer-actions')?.insertAdjacentHTML('beforebegin', `<div class="d25-drawer-route"><b>${isCurrent ? '⚔ Дружина находится здесь' : campaign.targetId === province.id ? '⚑ Цель действующего похода' : 'Маршрут действующего похода'}</b><span>${d25Esc(d25CampaignPhaseText(campaign))} · припасы ${round(campaign.supply)}</span><button data-tab="war" data-native-press>Открыть военный совет</button></div>`);
  }
  drawer.querySelectorAll('[data-campaign-target]').forEach((button) => button.addEventListener('click', () => d25StartCampaign(button.dataset.campaignTarget, state.campaignCommanderId)));
  drawer.querySelectorAll('[data-tab="war"]').forEach((button) => button.addEventListener('click', () => { currentTab = 'war'; closeDrawer(); render(); }));
};
