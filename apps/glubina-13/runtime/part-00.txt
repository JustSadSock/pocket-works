function readJSON(key, fallback) {
  try {
    const value = JSON.parse(localStorage.getItem(key));
    return value ?? fallback;
  } catch {
    return fallback;
  }
}

function writeJSON(key, value) {
  try {
    localStorage.setItem(key, JSON.stringify(value));
  } catch (error) {
    console.warn('ГЛУБИНА 13: не удалось сохранить данные', error);
  }
}

function angleLerp(from, to, amount) {
  let delta = ((to - from + Math.PI) % (Math.PI * 2)) - Math.PI;
  if (delta < -Math.PI) delta += Math.PI * 2;
  return from + delta * amount;
}

function restoreRun(saved) {
  const run = createRun(saved.seed);
  run.player.x = saved.player.x;
  run.player.z = saved.player.z;
  run.player.heading = saved.player.heading;
  run.oxygen = saved.oxygen;
  run.hull = Math.min(run.maxHull, saved.hull);
  run.sonar = saved.sonar;
  run.cargo = saved.cargo;
  run.collected = new Set(saved.collected);
  run.elapsed = saved.elapsed;
  run.distanceTravelled = saved.distanceTravelled;
  run.maxRange = saved.maxRange;
  run.attention = saved.attention;
  run.startedAt = saved.startedAt;
  if (saved.listener?.active) {
    run.listener = {
      active: true,
      x: saved.listener.x,
      z: saved.listener.z,
      heading: saved.listener.heading,
      phase: 0,
    };
  }
  return run;
}

function worldMesh(mesh, x, y, z) {
  mesh.world = [x, y, z];
  return mesh;
}

function removeMeshes(meshes) {
  for (const mesh of meshes) engine?.remove(mesh);
}

function clearWorld() {
  for (const entry of loadedChunks.values()) removeMeshes(entry.meshes);
  loadedChunks.clear();
  removeMeshes(listenerMeshes);
  listenerMeshes = [];
}

function createArchMeshes(entity, target) {
  const scale = entity.scale;
  const rotated = Math.abs(entity.rotation) > 0.1;
  const offsets = [-1.45 * scale, 1.45 * scale];
  for (const offset of offsets) {
    const mesh = engine.createMesh(geometry.box);
    const x = entity.x + (rotated ? 0 : offset);
    const z = entity.z + (rotated ? offset : 0);
    worldMesh(mesh, x, 0.25, z);
    mesh.scale = rotated ? [0.46 * scale, 2.3 * scale, 0.7 * scale] : [0.7 * scale, 2.3 * scale, 0.46 * scale];
    mesh.color = [...palette.stone];
    mesh.edgeVisibility = 0.65;
    target.push(mesh);
  }
  const beam = engine.createMesh(geometry.box);
  worldMesh(beam, entity.x, 2.55 * scale, entity.z);
  beam.scale = rotated ? [0.55 * scale, 0.42 * scale, 2.2 * scale] : [2.2 * scale, 0.42 * scale, 0.55 * scale];
  beam.color = [...palette.stone];
  beam.edgeVisibility = 0.7;
  target.push(beam);
}

function createTreasureMeshes(entity, target) {
  const type = entity.treasureType;
  const color = palette[type];
  const primary = engine.createMesh(type === 'archive' ? geometry.box : type === 'idol' ? geometry.tetra : geometry.octa);
  worldMesh(primary, entity.x, -0.85, entity.z);
  primary.scale = type === 'archive' ? [0.72, 0.28, 0.52] : type === 'idol' ? [0.72, 1.05, 0.72] : [0.62, 0.82, 0.62];
  primary.rotation = [0.18, entity.rotation, type === 'archive' ? 0.16 : 0];
  primary.color = [...color];
  primary.edgeVisibility = 1;
  primary.sonarGain = 1.55;
  primary.entity = entity;
  target.push(primary);

  if (type === 'idol') {
    const crown = engine.createMesh(geometry.octa);
    worldMesh(crown, entity.x, 0.2, entity.z);
    crown.scale = [0.38, 0.38, 0.38];
    crown.color = [...palette.idol];
    crown.edgeVisibility = 1;
    crown.sonarGain = 1.7;
    crown.entity = entity;
    target.push(crown);
  }
}

function createBaseMeshes(target) {
  const platform = engine.createMesh(geometry.cylinder);
  worldMesh(platform, 0, -1.72, 0);
  platform.scale = [3.6, 0.18, 3.6];
  platform.color = [0.26, 0.38, 0.38];
  platform.edgeVisibility = 0.7;
  platform.emissive = 0.035;
  platform.sonarGain = 1.2;
  target.push(platform);

  for (const [x, z] of [[-3.2,-2.4],[3.2,-2.4],[-3.2,2.4],[3.2,2.4]]) {
    const column = engine.createMesh(geometry.box);
    worldMesh(column, x, 0.35, z);
    column.scale = [0.42, 2.2, 0.42];
    column.color = [0.46, 0.52, 0.48];
    column.edgeVisibility = 0.85;
    column.emissive = 0.018;
    target.push(column);
  }

  const header = engine.createMesh(geometry.box);
  worldMesh(header, 0, 2.55, -2.4);
  header.scale = [3.65, 0.32, 0.42];
  header.color = [0.52, 0.57, 0.52];
  header.edgeVisibility = 0.8;
  header.emissive = 0.025;
  target.push(header);
}

function createWreckEntity() {
  const wreck = profile.wreck;
  if (!wreck || wreck.seed !== state.seed) return null;
  return {
    id: 'black-box:previous-expedition',
    type: 'wreck',
    x: wreck.x,
    z: wreck.z,
    item: wreck.item,
    rotation: 0.4,
  };
}
