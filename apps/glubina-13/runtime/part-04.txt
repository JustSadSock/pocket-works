  ensureAudio();
}

function releaseJoystick(event) {
  if (joystickPointer !== null && event?.pointerId !== undefined && event.pointerId !== joystickPointer) return;
  joystickPointer = null;
  joystickVector = { x: 0, y: 0, magnitude: 0 };
  joystickKnob.style.transform = 'translate(-50%, -50%)';
  joystick.dataset.active = 'false';
}

function openSettings() {
  settingsLayer.hidden = false;
  refreshSettings();
}

function closeSettings() {
  settingsLayer.hidden = true;
}

function refreshSettings() {
  for (const [button, enabled] of [[soundToggle, settings.sound], [hapticToggle, settings.haptics], [motionToggle, settings.motion]]) {
    button.dataset.enabled = enabled ? 'true' : 'false';
    button.querySelector('em').textContent = enabled ? 'Вкл' : 'Выкл';
  }
  bestValue.textContent = profile.bestValue;
}

function toggleSetting(key) {
  settings[key] = !settings[key];
  saveSettings();
  refreshSettings();
  if (key === 'sound' && settings.sound) ensureAudio();
}

  if (mode !== 'playing') return;
  event.preventDefault();
  joystickPointer = event.pointerId;
  joystick.setPointerCapture(event.pointerId);
  updateJoystick(event);
}, { passive: false });

  if (event.pointerId !== joystickPointer) return;
  event.preventDefault();
  updateJoystick(event);

window.addEventListener('keydown', (event) => {
  keys.add(event.code);
  if (event.code === 'Space') {
    event.preventDefault();
    fireSonar();
  }
  if (event.code === 'Escape') {
    if (!settingsLayer.hidden) closeSettings();
    else if (mode === 'playing') pauseGame();
    else if (mode === 'paused') continueGame();
  }
});
